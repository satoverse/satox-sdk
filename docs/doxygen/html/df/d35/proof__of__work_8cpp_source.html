96 unsigned char hash[SHA256_DIGEST_LENGTH];
97 SHA256_CTX sha256;
98 SHA256_Init(&sha256);
99 SHA256_Update(&sha256, input.c_str(), input.size());
100 SHA256_Final(hash, &sha256);

102 std::stringstream ss;
103 for(int i = 0; i < SHA256_DIGEST_LENGTH; i++) {
104 ss << std::hex << std::setw(2) << std::setfill('0') << static_cast<int>(hash[i]);

144bool ProofOfWork::meetsDifficulty(const std::string& hash, uint64_t difficulty) {
145 // Optimize difficulty check with SIMD
146 if (difficulty > hash.length() * 4) {
147 return false;
148 }
149
150 // Convert first few characters to binary using SIMD
151 __m256i zero = _mm256_setzero_si256();
152 __m256i ones = _mm256_set1_epi8(1);
153
154 int i = 0;
155 while (i + 32 <= difficulty) {
156 __m256i chars = _mm256_loadu_si256((__m256i*)&hash[i/4]);
157 __m256i result = _mm256_cmpeq_epi8(chars, zero);
158 if (!_mm256_testz_si256(result, ones)) {
159 return false;
160 }
161 i += 32;
162 }
163
164 // Handle remaining bits
165 while (i < difficulty) {
166 if (hash[i/4] != '0') {
167 return false;
168 }
169 i++;
170 }
171
172 return true;
173}

203bool ProofOfWork::isValidNonce(const std::string& nonce) {
204 if (nonce.length() != 32) {
205 return false;

207
208 // Optimize nonce validation with SIMD
209 __m256i valid_chars = _mm256_setr_epi8(
210 '0', '1', '2', '3', '4', '5', '6', '7',
211 '8', '9', 'a', 'b', 'c', 'd', 'e', 'f',
212 '0', '1', '2', '3', '4', '5', '6', '7',
213 '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'
214 );
215
216 for (int i = 0; i < 32; i += 32) {
217 __m256i chars = _mm256_loadu_si256((__m256i*)&nonce[i]);
218 __m256i result = _mm256_cmpeq_epi8(chars, valid_chars);
219 if (!_mm256_testz_si256(result, _mm256_set1_epi8(1))) {
220 return false;
