83void NetworkInterface::handleIncomingMessage(const NetworkMessage& message) {
84 if (!running_) {
85 spdlog::warn("Received message while NetworkInterface is not running");
86 return;
87 }
88
89 try {
90 spdlog::debug("Handling incoming message of type: {}",
91 static_cast<int>(message.type));
92
93 if (message.type == NetworkMessage::Type::SYNC_RESPONSE) {
94 handleResponse(message);
95 } else if (messageCallback_) {
96 messageCallback_(message);

98 spdlog::warn("Received message but no callback registered");
99 }
100 } catch (const std::exception& e) {
101 spdlog::error("Error handling message: {}", e.what());
102 if (errorCallback_) {
103 errorCallback_(e.what());
104 }

108void NetworkInterface::handleResponse(const NetworkMessage& response) {
109 if (!running_) {
110 spdlog::warn("Received response while NetworkInterface is not running");
111 return;
112 }
113
114 if (response.requestId.empty()) {
115 spdlog::warn("Received response with empty request ID");
116 return;
117 }
118
119 try {
120 std::lock_guard<std::mutex> lock(responseMutex_);
121 auto it = pendingResponses_.find(response.requestId);
122 if (it != pendingResponses_.end()) {
123 spdlog::debug("Found pending response for request ID: {}", response.requestId);
124 it->second.promise.set_value(response);
125 pendingResponses_.erase(it);
126 } else {
127 spdlog::warn("Received response for unknown request ID: {}", response.requestId);
128 }
129 } catch (const std::exception& e) {
130 spdlog::error("Error handling response: {}", e.what());
131 if (errorCallback_) {
132 errorCallback_(e.what());

137std::future<NetworkMessage> NetworkInterface::waitForResponse(
138 const std::string& requestId,
139 std::chrono::milliseconds timeout) {
140
141 if (requestId.empty()) {
142 spdlog::error("Cannot wait for response with empty request ID");
143 std::promise<NetworkMessage> promise;
144 NetworkMessage errorMsg;
145 errorMsg.type = NetworkMessage::Type::ERROR;
146 errorMsg.payload = "Invalid request ID";
147 promise.set_value(errorMsg);
148 return promise.get_future();
149 }
150
151 if (timeout.count() <= 0) {
152 spdlog::warn("Invalid timeout value: {}ms, using default 30s", timeout.count());
153 timeout = std::chrono::seconds(30);
154 }
155
156 std::future<NetworkMessage> future;
157 {
158 std::lock_guard<std::mutex> lock(responseMutex_);
159 auto it = pendingResponses_.find(requestId);
160 if (it == pendingResponses_.end()) {
161 spdlog::debug("Creating new pending response for request ID: {}", requestId);
162 PendingResponse pending;
163 pending.expiry = std::chrono::system_clock::now() + timeout;
164 future = pending.promise.get_future();
165 pendingResponses_[requestId] = std::move(pending);
166 } else {
167 spdlog::warn("Request ID already exists: {}", requestId);
168 future = it->second.promise.get_future();
169 }
170 }
171 return future;
172 }

174void NetworkInterface::cleanupExpiredResponses() {
175 auto now = std::chrono::system_clock::now();
176 std::lock_guard<std::mutex> lock(responseMutex_);
177
178 for (auto it = pendingResponses_.begin(); it != pendingResponses_.end();) {
179 if (it->second.expiry < now) {
180 NetworkMessage timeoutResponse;
181 timeoutResponse.type = NetworkMessage::Type::ERROR;
182 timeoutResponse.payload = "Response timeout";
183 it->second.promise.set_value(timeoutResponse);
184 it = pendingResponses_.erase(it);
185 } else {
186 ++it;
187 }
188 }

207void NetworkInterface::disconnect() {
208 running_ = false;
209 if (messageThread_.joinable()) {
210 messageThread_.join();
211 }
212 connected_ = false;
213
214 // Reject all pending responses
215 std::lock_guard<std::mutex> lock(responseMutex_);
216 for (auto& [requestId, pending] : pendingResponses_) {
217 NetworkResponse response;
218 response.success = false;
219 response.error = "Connection closed";
220 pending.promise.set_value(response);
221 }
222 pendingResponses_.clear();
223}

229void NetworkInterface::sendMessage(const NetworkMessage& message) {
230 if (!running_) {
231 spdlog::error("Cannot send message: NetworkInterface is not running");
232 if (errorCallback_) {
233 errorCallback_("NetworkInterface is not running");
234 }
235 return;
236 }
237
238 // Validate message
239 if (message.type == NetworkMessage::Type::SYNC_REQUEST && message.requestId.empty()) {
240 spdlog::error("SYNC_REQUEST message must have a request ID");
241 if (errorCallback_) {
242 errorCallback_("SYNC_REQUEST message must have a request ID");
243 }
244 return;
245 }
246
247 try {
248 std::lock_guard<std::mutex> lock(queueMutex_);
249 messageQueue_.push(message);
250 queueCondition_.notify_one();
251 spdlog::debug("Message queued for sending, type: {}, requestId: {}",
252 static_cast<int>(message.type), message.requestId);
253 } catch (const std::exception& e) {
254 spdlog::error("Failed to queue message: {}", e.what());
255 if (errorCallback_) {
256 errorCallback_(std::string("Failed to queue message: ") + e.what());
257}
258 }

261void NetworkInterface::processMessageQueue() {
262 if (!running_) {
263 return;
264 }
265
266 try {
267 std::unique_lock<std::mutex> lock(queueMutex_);
268 if (messageQueue_.empty()) {
269 return;
270 }
271
272 // Process messages in batches to avoid holding the lock too long
273 const size_t MAX_BATCH_SIZE = 10;
274 size_t processed = 0;
275
276 while (!messageQueue_.empty() && processed < MAX_BATCH_SIZE) {
277 NetworkMessage message = messageQueue_.front();
278 messageQueue_.pop();
279 lock.unlock();
280
281 try {
282 spdlog::debug("Processing message from queue, type: {}",
283 static_cast<int>(message.type));
284 handleIncomingMessage(message);
285 } catch (const std::exception& e) {
286 spdlog::error("Error processing message: {}", e.what());
287 if (errorCallback_) {
288 errorCallback_(e.what());
289}
290 }
291
292 lock.lock();
293 processed++;
294 }
295 } catch (const std::exception& e) {
296 spdlog::error("Error in message queue processing: {}", e.what());
297 if (errorCallback_) {
298 errorCallback_(e.what());
299 }
300 }
