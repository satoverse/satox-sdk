75bool ConfigManager::loadConfig(const std::string& path) {
76 std::lock_guard<std::mutex> lock(mutex_);
77
78 if (!std::filesystem::exists(path)) {
79 lastError_ = "Configuration file does not exist: " + path;
80 return false;
81 }
82
83 std::ifstream file(path);
84 if (!file.is_open()) {
85 lastError_ = "Failed to open configuration file: " + path;
86 return false;
87 }
88
89 std::stringstream buffer;
90 buffer << file.rdbuf();
91 std::string content = buffer.str();
92
93 if (!validateConfigFile(content)) {
94 return false;
95 }
96
97 if (!parseConfigFile(content)) {
98 return false;
99 }
100
101 configPath_ = path;
102 updateStats(true);
103 spdlog::info("Config loaded from: {}", path);
104 return true;
105}

457bool ConfigManager::loadFromEnvironment(const std::string& prefix) {
458 std::lock_guard<std::mutex> lock(mutex_);
459
460 for (const auto& [key, value] : defaultValues_) {
461 std::string envVar = prefix + key;
462 std::transform(envVar.begin(), envVar.end(), envVar.begin(),
463 [](unsigned char c) { return std::toupper(c); });
464
465 std::string envValue = getEnvironmentVariable(envVar);
466 if (!envValue.empty()) {
467 size_t dotPos = key.find('.');
468 if (dotPos != std::string::npos) {
469 std::string section = key.substr(0, dotPos);
470 std::string configKey = key.substr(dotPos + 1);
471 setValue(section, configKey, value);
472 }
473 }
474 }
475
476 return true;
477}

573bool ConfigManager::parseConfigFile(const std::string& content) {
574 try {
575 nlohmann::json config = nlohmann::json::parse(content);
576
577 for (const auto& [sectionName, sectionData] : config.items()) {
578 if (!validateSection(sectionName)) {
579 lastError_ = "Invalid section name: " + sectionName;
580 return false;
581 }
582
583 ConfigSection configSection;
584 configSection.name = sectionName;
585
586 for (const auto& [key, value] : sectionData.items()) {
587 if (!validateKey(key)) {
588 lastError_ = "Invalid key name: " + key;
589 return false;
590 }
591
592 configSection.values[key] = createValue(value);
593 }
594
595 sections_[sectionName] = configSection;
596 }
597
598 return true;
599 } catch (const std::exception& e) {
600 lastError_ = "Failed to parse config file: " + std::string(e.what());
601 return false;
602 }
603}

658ConfigManager::ConfigValue ConfigManager::createValue(const nlohmann::json& json) const {
659 ConfigValue value;
660
661 if (json.is_string()) {
662 value.type = ValueType::STRING;
663 value.stringValue = json.get<std::string>();
664 } else if (json.is_number_integer()) {
665 value.type = ValueType::INTEGER;
666 value.intValue = json.get<int64_t>();
667 } else if (json.is_number_float()) {
668 value.type = ValueType::FLOAT;
669 value.floatValue = json.get<double>();
670 } else if (json.is_boolean()) {
671 value.type = ValueType::BOOLEAN;
672 value.boolValue = json.get<bool>();
673 } else if (json.is_array()) {
674 value.type = ValueType::ARRAY;
675 value.complexValue = json;
676 } else if (json.is_object()) {
677 value.type = ValueType::OBJECT;
678 value.complexValue = json;
679 } else if (json.is_null()) {
680 value.type = ValueType::NULL_VALUE;
681 }
682
683 return value;
684}
