459void AssetManager::updateAssetStats(const std::string& id, AssetState state) {
460 if (!statsEnabled_) {
461 return;
462 }
463
464 stats_.totalAssets = assets_.size();
465 stats_.issuedAssets = std::count_if(assets_.begin(), assets_.end(),
466 [](const auto& pair) { return pair.second.state == AssetState::ISSUED; });
467 stats_.transferredAssets = std::count_if(assets_.begin(), assets_.end(),
468 [](const auto& pair) { return pair.second.state == AssetState::TRANSFERRED; });
469 stats_.burnedAssets = std::count_if(assets_.begin(), assets_.end(),
470 [](const auto& pair) { return pair.second.state == AssetState::BURNED; });
471 stats_.errorAssets = std::count_if(assets_.begin(), assets_.end(),
472 [](const auto& pair) { return pair.second.state == AssetState::ERROR; });
473
474 // Update type distribution
475 stats_.typeDistribution = nlohmann::json::object();
476 for (const auto& [id, asset] : assets_) {
477 std::string typeStr = std::to_string(static_cast<int>(asset.metadata.type));
478 stats_.typeDistribution[typeStr] = stats_.typeDistribution.value(typeStr, 0) + 1;
479 }
480}
