59bool LoggingManager::initialize(const LogConfig& config) {
60 std::lock_guard<std::mutex> lock(mutex_);
61
62 if (initialized_) {
63 lastError_ = "Logging manager already initialized";
64 return false;
65 }
66
67 if (!validateConfig(config)) {
68 return false;
69 }
70
71 config_ = config;
72 stats_ = LogStats{};
73 stats_.lastFlush = std::chrono::system_clock::now();
74
75 // Create log directory if it doesn't exist
76 if (config_.fileOutput) {
77 std::filesystem::create_directories(config_.logDir);
78 logFile_.open(config_.logDir + "/" + config_.logFile, std::ios::app);
79 if (!logFile_.is_open()) {
80 lastError_ = "Failed to open log file";
81 return false;
82 }
83 }
84
85 // Start async logging worker if enabled
86 if (config_.asyncLogging) {
87 running_ = true;
88 workerThread_ = std::thread(&LoggingManager::asyncLoggingWorker, this);
89 }
90
91 initialized_ = true;
92 return true;
93}

114void LoggingManager::log(Level level, const std::string& message,
115 const std::string& source,
116 const nlohmann::json& metadata) {
117 if (!initialized_ || level < config_.minLevel) {
118 return;
119 }
120
121 LogEntry entry;
122 entry.level = level;
123 entry.message = message;
124 entry.source = source;
125 entry.threadId = getThreadId();
126 entry.timestamp = std::chrono::system_clock::now();
127 entry.metadata = metadata;
128
129 if (config_.asyncLogging) {
130 std::lock_guard<std::mutex> lock(mutex_);
131 if (logQueue_.size() < config_.queueSize) {
132 logQueue_.push(entry);
133 stats_.queueSize = logQueue_.size();
134 } else {
135 stats_.droppedEntries++;
136 }
137 } else {
138 processLogEntry(entry);
139 }
140}

192void LoggingManager::flush() {
193 std::lock_guard<std::mutex> lock(mutex_);
194
195 if (config_.asyncLogging) {
196 while (!logQueue_.empty()) {
197 processLogEntry(logQueue_.front());
198 logQueue_.pop();
199 }
200 stats_.queueSize = 0;
201 }
202
203 if (logFile_.is_open()) {
204 logFile_.flush();
205 }
206
207 stats_.lastFlush = std::chrono::system_clock::now();
208}

210void LoggingManager::rotate() {
211 std::lock_guard<std::mutex> lock(mutex_);
212
213 if (!config_.fileOutput) {
214 return;
215 }
216
217 // Close current log file
218 if (logFile_.is_open()) {
219 logFile_.close();
220 }
221
222 // Rotate existing log files
223 for (size_t i = config_.maxFiles - 1; i > 0; --i) {
224 std::string oldFile = config_.logDir + "/" + config_.logFile + "." + std::to_string(i);
225 std::string newFile = config_.logDir + "/" + config_.logFile + "." + std::to_string(i + 1);
226 std::filesystem::rename(oldFile, newFile);
227 }
228
229 // Rename current log file
230 std::string currentFile = config_.logDir + "/" + config_.logFile;
231 std::string newFile = config_.logDir + "/" + config_.logFile + ".1";
232 std::filesystem::rename(currentFile, newFile);
233
234 // Open new log file
235 logFile_.open(currentFile, std::ios::app);
236}

287std::string LoggingManager::formatLogEntry(const LogEntry& entry) {
288 std::stringstream ss;
289
290 if (config_.includeTimestamp) {
291 auto time = std::chrono::system_clock::to_time_t(entry.timestamp);
292 ss << std::put_time(std::localtime(&time), "%Y-%m-%d %H:%M:%S") << " ";
293 }
294
295 ss << "[" << levelToString(entry.level) << "] ";
296
297 if (config_.includeThreadId) {
298 ss << "[" << entry.threadId << "] ";
299 }
300
301 if (config_.includeSource && !entry.source.empty()) {
302 ss << "[" << entry.source << "] ";
303 }
304
305 ss << entry.message;
306
307 if (!entry.metadata.empty()) {
308 ss << " " << entry.metadata.dump();
309 }
310
311 return ss.str();
312}

324void LoggingManager::asyncLoggingWorker() {
325 while (running_) {
326 LogEntry entry;
327 bool hasEntry = false;
328
329 {
330 std::lock_guard<std::mutex> lock(mutex_);
331 if (!logQueue_.empty()) {
332 entry = logQueue_.front();
333 logQueue_.pop();
334 stats_.queueSize = logQueue_.size();
335 hasEntry = true;
336 }
337 }
338
339 if (hasEntry) {
340 processLogEntry(entry);
341 }
342
343 // Check if it's time to flush
344 auto now = std::chrono::system_clock::now();
345 if (now - stats_.lastFlush >= config_.flushInterval) {
346 flush();
347 }
348
349 std::this_thread::sleep_for(std::chrono::milliseconds(10));
350 }
351}

386LoggingManager::Level LoggingManager::stringToLevel(const std::string& level) const {
387 std::string upperLevel = level;
388 std::transform(upperLevel.begin(), upperLevel.end(), upperLevel.begin(), ::toupper);
389
390 if (upperLevel == "TRACE") return Level::TRACE;
391 if (upperLevel == "DEBUG") return Level::DEBUG;
392 if (upperLevel == "INFO") return Level::INFO;
393 if (upperLevel == "WARNING") return Level::WARNING;
394 if (upperLevel == "ERROR") return Level::ERROR;
395 if (upperLevel == "FATAL") return Level::FATAL;
