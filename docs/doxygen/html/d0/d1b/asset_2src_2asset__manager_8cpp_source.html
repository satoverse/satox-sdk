80bool AssetManager::initialize(const AssetConfig& config) {
81 std::lock_guard<std::mutex> lock(mutex_);
82
83 if (initialized_) {
84 setLastError("AssetManager already initialized");
85 return false;
86 }
87
88 try {
89 SimpleLogger::info("AssetManager: Initializing with config: " + config.name);
90
91 // Validate configuration
92 if (!validateConfig(config)) {
93 setLastError("Invalid configuration");
94 return false;
95 }
96
97 // Set state to initializing
98 state_ = AssetState::INITIALIZING;
99
100 // Store configuration
101 config_ = config;
102
103 // Initialize logging if enabled
104 if (config.enableLogging && !config.logPath.empty()) {
105 try {
106 std::filesystem::create_directories(
107 std::filesystem::path(config.logPath).parent_path()
108 );
109 SimpleLogger::info("Logging initialized to: " + config.logPath);
110 } catch (const std::exception& e) {
111 SimpleLogger::warn("Failed to setup logging: " + std::string(e.what()));
112 }
113 }
114
115 // Initialize asset storage
116 assets_.clear();
117 balances_.clear();
118 assetStatuses_.clear();
119
120 // Set state to initialized
121 state_ = AssetState::INITIALIZED;
122 initialized_ = true;
123 lastHealthCheck_ = std::chrono::system_clock::now();
124
125 SimpleLogger::info("AssetManager: Initialized successfully");
126 notifyAssetEvent("initialized", true);
127 return true;
128
129 } catch (const std::exception& e) {
130 SimpleLogger::error("AssetManager: Initialization failed: " + std::string(e.what()));
131 setLastError(e.what());
132 state_ = AssetState::ERROR;
133 return false;
134 }

181bool AssetManager::createAsset(const AssetCreationRequest& request) {
182 std::lock_guard<std::mutex> lock(mutex_);
183
184 if (!initialized_) {
185 setLastError("AssetManager not initialized");
186 return false;
187 }
188
189 try {
190 auto startTime = std::chrono::high_resolution_clock::now();
191
192 // Validate request
193 AssetVerificationResult result;
194 if (!validateAssetCreationRequest(request, result)) {
195 setLastError(result.error_message);
196 updateStats(false, 0.0);
197 notifyErrorEvent("createAsset", result.error_message);
198 notifyAssetEvent("asset_creation_failed", false);
199 AssetResult opResult{"createAsset", false, result.error_message};
200 notifyOperationEvent(opResult);
201 return false;
202 }
203
204 // Check if asset already exists
205 if (assets_.find(request.name) != assets_.end()) {
206 setLastError("Asset already exists: " + request.name);
207 updateStats(false, 0.0);
208 notifyErrorEvent("createAsset", "Asset already exists: " + request.name);
209 return false;
210 }
211
212 // Check max assets limit
213 if (assets_.size() >= config_.maxAssets) {
214 setLastError("Maximum number of assets reached: " + std::to_string(config_.maxAssets));
215 updateStats(false, 0.0);
216 notifyErrorEvent("createAsset", "Maximum number of assets reached: " + std::to_string(config_.maxAssets));
217 return false;
218 }
219
220 // Validate metadata size
221 if (request.metadata.is_object() && request.metadata.size() > 100) {
222 setLastError("Metadata too large, maximum 100 fields allowed");
223 updateStats(false, 0.0);
224 notifyErrorEvent("createAsset", "Metadata too large, maximum 100 fields allowed");
225 return false;
226 }
227
228 // Create asset metadata
229 AssetMetadata metadata;
230 metadata.name = request.name;
231 metadata.symbol = request.symbol;
232 metadata.description = request.metadata.is_object() && request.metadata.contains("description")
233 ? request.metadata["description"].get<std::string>() : "";
234 metadata.issuer = request.owner_address;
235 metadata.total_supply = request.amount;
236 metadata.decimals = request.units;
237 metadata.reissuable = request.reissuable;
238 metadata.frozen = false;
239 metadata.additional_data = request.metadata;
240
241 // Store asset
242 assets_[request.name] = metadata;
243 assetStatuses_[request.name] = AssetStatus::ACTIVE;
244
245 // Initialize balance for owner
246 balances_[request.name][request.owner_address] = request.amount;
247
248 auto endTime = std::chrono::high_resolution_clock::now();
249 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(endTime - startTime);
250
251 updateStats(true, duration.count());
252 logOperation("createAsset", true, duration.count());
253
254 SimpleLogger::info("Asset created: " + request.name + " (" + request.symbol + ")");
255 notifyAssetEvent("asset_created", true);
256
257 AssetResult opResult{"createAsset", true, ""};
258 notifyOperationEvent(opResult);
259
260 return true;
261
262 } catch (const std::exception& e) {
263 SimpleLogger::error("Asset creation failed: " + std::string(e.what()));
264 setLastError(e.what());
265 updateStats(false, 0.0);
266 notifyErrorEvent("createAsset", e.what());
267 return false;
268 }

271bool AssetManager::transferAsset(const AssetTransferRequest& request) {
272 std::lock_guard<std::mutex> lock(mutex_);
273
274 if (!initialized_) {
275 setLastError("AssetManager not initialized");
276 return false;
277 }
278
279 try {
280 auto startTime = std::chrono::high_resolution_clock::now();
281
282 // Validate transfer
283 AssetVerificationResult result;
284 if (!validateAssetTransfer(request, result)) {
285 setLastError(result.error_message);
286 updateStats(false, 0.0);
287 notifyErrorEvent("transferAsset", result.error_message);
288 return false;
289 }
290
291 // Check if asset exists
292 if (assets_.find(request.asset_name) == assets_.end()) {
293 setLastError("Asset not found: " + request.asset_name);
294 updateStats(false, 0.0);
295 notifyErrorEvent("transferAsset", "Asset not found: " + request.asset_name);
296 return false;
297 }
298
299 // Check if asset is frozen
300 if (assetStatuses_[request.asset_name] == AssetStatus::FROZEN) {
301 setLastError("Asset is frozen: " + request.asset_name);
302 updateStats(false, 0.0);
303 notifyErrorEvent("transferAsset", "Asset is frozen: " + request.asset_name);
304 return false;
305 }
306
307 // Check sender balance
308 auto& assetBalances = balances_[request.asset_name];
309 if (assetBalances[request.from_address] < request.amount) {
310 setLastError("Insufficient balance for transfer");
311 updateStats(false, 0.0);
312 notifyErrorEvent("transferAsset", "Insufficient balance for transfer");
313 return false;
314 }
315
316 // Perform transfer
317 assetBalances[request.from_address] -= request.amount;
318 assetBalances[request.to_address] += request.amount;
319
320 auto endTime = std::chrono::high_resolution_clock::now();
321 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(endTime - startTime);
322
323 updateStats(true, duration.count());
324 logOperation("transferAsset", true, duration.count());
325
326 SimpleLogger::info("Asset transferred: " + request.asset_name + " from " + request.from_address + " to " + request.to_address + " (amount: " + std::to_string(request.amount) + ")");
327 notifyAssetEvent("asset_transferred", true);
328
329 return true;
330
331 } catch (const std::exception& e) {
332 SimpleLogger::error("Asset transfer failed: " + std::string(e.what()));
333 setLastError(e.what());
334 updateStats(false, 0.0);
335 notifyErrorEvent("transferAsset", e.what());
336 return false;
337 }

947bool AssetManager::validateAssetCreationRequest(const AssetCreationRequest& request, AssetVerificationResult& result) const {
948 result.valid = true;
949 result.error_message.clear();
950 result.warnings.clear();
951
952 // Validate required fields
953 if (request.name.empty()) {
954 result.valid = false;
955 result.error_message = "Asset name cannot be empty";
956 return false;
957 }
958
959 if (request.symbol.empty()) {
960 result.valid = false;
961 result.error_message = "Asset symbol cannot be empty";
962 return false;
963 }
964
965 if (request.amount == 0) {
966 result.valid = false;
967 result.error_message = "Asset amount must be greater than 0";
968 return false;
969 }
970
971 if (request.owner_address.empty()) {
972 result.valid = false;
973 result.error_message = "Owner address cannot be empty";
974 return false;
975 }
976
977 // Validate units (decimals)
978 if (request.units > 18) {
979 result.warnings.push_back("Asset units should not exceed 18");
980 }
981
982 return true;
983}

1028 } else {

1033
1034 // Update average operation time
