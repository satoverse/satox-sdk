62 std::lock_guard<std::mutex> lock(mutex_);
63 if (initialized_) {
64 return true;
65 }
66
67 if (!std::filesystem::exists(configPath)) {
68 lastError_ = {1, "Configuration file not found"};
69 return false;
70 }
71
72 try {
73 std::ifstream configFile(configPath);
74 nlohmann::json config = nlohmann::json::parse(configFile);

76 // Load node configurations
77 if (config.contains("nodes")) {
78 for (const auto& node : config["nodes"]) {

82
83 // Load bandwidth limit if specified

85 bandwidthLimit_ = config["bandwidthLimit"];

87
88 configPath_ = configPath;
89 initialized_ = true;
90 return true;
91 } catch (const std::exception& e) {
92 lastError_ = {2, std::string("Failed to initialize: ") + e.what()};

106std::future<bool> ContentDistribution::distributeContent(const std::string& contentHash, const std::vector<std::string>& targetNodes) {
107 return std::async(std::launch::async, [this, contentHash, targetNodes]() {
108 if (!initialized_) {
109 lastError_ = {3, "Content Distribution not initialized"};
110 return false;
111 }
112
113 // Initialize distribution status
114 DistributionStatus status;
115 status.contentHash = contentHash;
116 status.status = "pending";
117 status.progress = 0;
118 status.startTime = getCurrentTimestamp();
119
120 if (!updateDistributionStatus(contentHash, status)) {
121 return false;
122 }
123
124 // Validate target nodes
125 std::vector<std::string> validNodes;
126 for (const auto& nodeId : targetNodes) {
127 if (isNodeAvailable(nodeId)) {
128 validNodes.push_back(nodeId);
129 }
130 }
131
132 if (validNodes.empty()) {
133 status.status = "failed";
134 status.error = "No valid target nodes available";
135 status.endTime = getCurrentTimestamp();
136 updateDistributionStatus(contentHash, status);
137 return false;
138 }
139
140 // Start distribution process
141 status.status = "in_progress";
142 updateDistributionStatus(contentHash, status);
143
144 // Simulate distribution process
145 for (const auto& nodeId : validNodes) {
146 // Check if distribution was cancelled
147 {
148 std::lock_guard<std::mutex> lock(mutex_);
149 auto it = activeDistributions_.find(contentHash);
150 if (it == activeDistributions_.end() || it->second.status == "cancelled") {
151 return false;
152 }
153 }
154
155 // Simulate node transfer
156 std::this_thread::sleep_for(std::chrono::milliseconds(100));
157
158 // Update progress
159 status.completedNodes.push_back(nodeId);
160 status.progress = (status.completedNodes.size() * 100) / validNodes.size();
161 updateDistributionStatus(contentHash, status);
162 }
163
164 // Mark distribution as completed
165 status.status = "completed";
166 status.progress = 100;
167 status.endTime = getCurrentTimestamp();
168 updateDistributionStatus(contentHash, status);
169
170 return true;
171 });
172}

313bool ContentDistribution::cancelDistribution(const std::string& contentHash) {
314 std::lock_guard<std::mutex> lock(mutex_);
315 auto it = activeDistributions_.find(contentHash);
316 if (it == activeDistributions_.end()) {
317 lastError_ = {8, "Distribution not found"};
318 return false;
319 }
320
321 if (it->second.status == "completed" || it->second.status == "failed") {
322 lastError_ = {9, "Cannot cancel completed or failed distribution"};
323 return false;
324 }

326 it->second.status = "cancelled";
