83bool CoreManager::initialize(const CoreConfig& config) {
84 std::lock_guard<std::mutex> lock(mutex_);
85 is_running_ = false; // Defensive: always reset before initialization
86 spdlog::debug("CoreManager::initialize called. is_running_ before: {}", is_running_);
87
88 if (initialized_) {
89 lastError_ = "CoreManager already initialized";
90 return false;
91 }
92
93 if (!validateConfig(config)) {
94 lastError_ = "Invalid configuration";
95 return false;
96 }
97
98 try {
99 config_ = config;
100 stats_.start_time = std::chrono::system_clock::now();
101
102 if (!initializeComponents()) {
103 lastError_ = "Failed to initialize components";
104 return false;
105 }
106
107 initialized_ = true;
108 spdlog::debug("CoreManager::initialize completed. is_running_: {}", is_running_);
109 return true;
110 } catch (const std::exception& e) {
111 lastError_ = e.what();
112 return false;
113 }
114}

116void CoreManager::shutdown() {
117 std::lock_guard<std::mutex> lock(mutex_);
118 spdlog::debug("CoreManager::shutdown called. is_running_ before: {}", is_running_);
119 // Always reset state for test isolation
120 spdlog::drop("core_manager");
121 shutdownComponents();
122 is_running_ = false;
123 spdlog::debug("CoreManager::shutdown sets is_running_ = false");
124 initialized_ = false;
125 component_status_.clear();
126 errorCallbacks_.clear();
127 statusCallbacks_.clear();
128 shutdownCallbacks_.clear();
129 lastError_.clear();
130 config_ = CoreConfig{};
131 spdlog::debug("CoreManager::shutdown completed. is_running_: {}", is_running_);
132 notifyShutdown();
133}

280bool CoreManager::restartComponent(const std::string& component) {
281 std::lock_guard<std::mutex> lock(mutex_);
282
283 if (!is_running_) {
284 lastError_ = "CoreManager not running";
285 return false;
286 }
287
288 try {
289 // Update component status to restarting
290 ComponentStatus status;
291 status.name = component;
292 status.state = ComponentState::RESTARTING;
293 status.last_update = std::chrono::system_clock::now();
294 component_status_[component] = status;
295
296 // Notify status change
297 notifyStatusChange(component, ComponentState::RESTARTING);
298
299 // Perform component-specific restart logic
300 bool success = false;
301 if (component == "network") {
302 auto& network = getNetworkManager();
303 network.shutdown();
304 success = network.initialize(config_.network_config);
305 } else if (component == "blockchain") {
306 auto& blockchain = getBlockchainManager();
307 blockchain.shutdown();
308 success = blockchain.initialize(config_.blockchain);
309 } else if (component == "database") {
310 // TODO: Implement proper interface for database operations without direct dependency
311 success = false; // Placeholder return, actual implementation needed
312 } else if (component == "wallet") {
313 auto& wallet = getWalletManager();
314 wallet.shutdown();
315 success = wallet.initialize(config_.wallet);
316 } else if (component == "asset") {
317 auto& asset = getAssetManager();
318 asset.shutdown();
319 success = asset.initialize(config_.asset);
320 } else if (component == "ipfs") {
321 auto& ipfs = getIPFSManager();
322 ipfs.shutdown();
323 success = ipfs.initialize(config_.ipfs);
324 } else {
325 lastError_ = "Unknown component: " + component;
326 return false;
327 }
328
329 // Update component status
330 status.state = success ? ComponentState::RUNNING : ComponentState::ERROR;
331 status.last_update = std::chrono::system_clock::now();
332 if (!success) {
333 status.error_count++;
334 }
335 component_status_[component] = status;
336
337 // Notify final status
338 notifyStatusChange(component, status.state);
339
340 return success;
341 } catch (const std::exception& e) {
342 lastError_ = "Failed to restart component: " + std::string(e.what());
343 return false;
344 }

442bool CoreManager::backupConfig(const std::string& backup_path) const {
443 std::lock_guard<std::mutex> lock(mutex_);
444
445 if (!is_running_) {
446 return false;
447 }
448
449 try {
450 // Create backup directory if it doesn't exist
451 std::filesystem::path backup_dir = std::filesystem::path(backup_path).parent_path();
452 if (!std::filesystem::exists(backup_dir)) {
453 std::filesystem::create_directories(backup_dir);
454 }
455
456 // Serialize configuration to JSON
457 nlohmann::json config_json;
458 config_json["data_dir"] = config_.data_dir;
459 config_json["network"] = config_.network_config;
460 config_json["enable_mining"] = config_.enable_mining;
461 config_json["enable_sync"] = config_.enable_sync;
462 config_json["sync_interval_ms"] = config_.sync_interval_ms;
463 config_json["mining_threads"] = config_.mining_threads;
464 config_json["rpc_endpoint"] = config_.rpc_endpoint;
465 config_json["rpc_username"] = config_.rpc_username;
466 config_json["rpc_password"] = config_.rpc_password;
467 config_json["timeout_ms"] = config_.timeout_ms;
468 config_json["backup_timestamp"] = std::chrono::system_clock::now().time_since_epoch().count();
469
470 // Write to file
471 std::ofstream file(backup_path);
472 if (!file.is_open()) {
473 return false;
474 }
475 file << config_json.dump(4);
476 file.close();
477
478 return true;
479 } catch (const std::exception& e) {
480 return false;
481 }

484bool CoreManager::restoreConfig(const std::string& backup_path) {
485 std::lock_guard<std::mutex> lock(mutex_);
486
487 try {
488 // Check if backup file exists
489 if (!std::filesystem::exists(backup_path)) {
490 lastError_ = "Backup file not found: " + backup_path;
491 return false;
492 }
493
494 // Read backup file
495 std::ifstream file(backup_path);
496 if (!file.is_open()) {
497 lastError_ = "Failed to open backup file: " + backup_path;
498 return false;
499 }
500
501 nlohmann::json config_json;
502 file >> config_json;
503 file.close();
504
505 // Validate backup data
506 if (!config_json.contains("data_dir") || !config_json.contains("network")) {
507 lastError_ = "Invalid backup file format";
508 return false;
509 }
510
511 // Restore configuration
512 CoreConfig restored_config;
513 restored_config.data_dir = config_json["data_dir"];
514 restored_config.network = config_json["network"];
515 restored_config.enable_mining = config_json.value("enable_mining", false);
516 restored_config.enable_sync = config_json.value("enable_sync", true);
517 restored_config.sync_interval_ms = config_json.value("sync_interval_ms", 1000);
518 restored_config.mining_threads = config_json.value("mining_threads", 1);
519 restored_config.rpc_endpoint = config_json.value("rpc_endpoint", "");
520 restored_config.rpc_username = config_json.value("rpc_username", "");
521 restored_config.rpc_password = config_json.value("rpc_password", "");
522 restored_config.timeout_ms = config_json.value("timeout_ms", 30000);
523
524 // Validate restored configuration
525 if (!validateConfig(restored_config)) {
526 lastError_ = "Invalid configuration in backup file";
527 return false;
528 }
529
530 // Apply restored configuration
531 config_ = restored_config;
532
533 // Update all components with new configuration
534 if (is_running_) {
535 updateAllComponents();
536 }
537
538 return true;
539 } catch (const std::exception& e) {
540 lastError_ = e.what();
541 return false;
542 }
