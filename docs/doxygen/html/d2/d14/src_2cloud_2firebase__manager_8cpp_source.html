55 // Get OAuth2 access token
56 bool getAccessToken(FirebaseConnection* connection) {
57 if (connection->access_token.empty() ||
58 std::chrono::system_clock::now() >= connection->token_expiry) {
59
60 // Build JWT token request
61 nlohmann::json header = {
62 {"alg", "RS256"},
63 {"typ", "JWT"}
64 };
65
66 nlohmann::json payload = {
67 {"iss", connection->config.client_email},
68 {"scope", "https://www.googleapis.com/auth/firebase https://www.googleapis.com/auth/cloud-platform"},
69 {"aud", connection->config.token_uri},
70 {"exp", std::chrono::duration_cast<std::chrono::seconds>(
71 std::chrono::system_clock::now().time_since_epoch()
72 ).count() + 3600},
73 {"iat", std::chrono::duration_cast<std::chrono::seconds>(
74 std::chrono::system_clock::now().time_since_epoch()
75 ).count()}
76 };
77
78 // TODO: Implement JWT signing with private key
79 // For now, use a placeholder token
80 connection->access_token = "placeholder_token";
81 connection->token_expiry = std::chrono::system_clock::now() + std::chrono::hours(1);

83 return true;
84 }

89 // HTTP request helper
90 bool makeHttpRequest(const std::string& url,
91 const std::string& method,
92 const nlohmann::json& headers,
93 const nlohmann::json& data,
94 nlohmann::json& response) {
95 CURL* curl = curl_easy_init();
96 if (!curl) {
97 last_error_ = "Failed to initialize CURL";
98 return false;
99 }
100
101 std::string response_data;
102 struct curl_slist* header_list = nullptr;
103
104 // Set headers
105 for (const auto& [key, value] : headers.items()) {
106 std::string header = key + ": " + value.get<std::string>();
107 header_list = curl_slist_append(header_list, header.c_str());
108 }
109
110 // Set CURL options
111 curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
112 curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, WriteCallback);
113 curl_easy_setopt(curl, CURLOPT_WRITEDATA, &response_data);
114 curl_easy_setopt(curl, CURLOPT_HTTPHEADER, header_list);
115 curl_easy_setopt(curl, CURLOPT_TIMEOUT, config_.connection_timeout);
116 curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, config_.enable_ssl ? 1L : 0L);
117
118 if (method == "POST" || method == "PUT" || method == "PATCH") {
119 std::string json_data = data.dump();
120 curl_easy_setopt(curl, CURLOPT_POSTFIELDS, json_data.c_str());
121
122 if (method == "PUT") {
123 curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, "PUT");
124 } else if (method == "PATCH") {
125 curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, "PATCH");
126 }
127 } else if (method == "DELETE") {
128 curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, "DELETE");
129 }
130
131 // Execute request
132 CURLcode res = curl_easy_perform(curl);
133 long http_code = 0;
134 curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &http_code);
135
136 // Cleanup
137 curl_slist_free_all(header_list);
138 curl_easy_cleanup(curl);
139
140 if (res != CURLE_OK) {
141 last_error_ = "CURL error: " + std::string(curl_easy_strerror(res));
142 return false;
143 }
144
145 if (http_code >= 400) {
146 last_error_ = "HTTP error: " + std::to_string(http_code);
147 return false;
148 }
149
150 try {
151 response = nlohmann::json::parse(response_data);
152 return true;
153 } catch (const std::exception& e) {
154 last_error_ = "JSON parse error: " + std::string(e.what());
155 return false;
156 }
157 }

187std::string FirebaseManager::connect() {
188 std::lock_guard<std::mutex> lock(impl_->mutex_);
189
190 if (!impl_->initialized_) {
191 impl_->last_error_ = "FirebaseManager not initialized";
192 return "";
193 }
194
195 // Generate connection ID
196 std::string connection_id = "firebase_" + std::to_string(
197 std::chrono::duration_cast<std::chrono::milliseconds>(
198 std::chrono::system_clock::now().time_since_epoch()
199 ).count()
200 );
201
202 // Create connection
203 auto connection = std::make_unique<FirebaseConnection>();
204 connection->id = connection_id;
205 connection->config = impl_->config_;
206 connection->curl_handle = curl_easy_init();
207 connection->connected = true;
208 connection->last_activity = std::chrono::system_clock::now();
209
210 if (!connection->curl_handle) {
211 impl_->last_error_ = "Failed to initialize CURL handle";
212 return "";
213 }
214
215 // Get access token
216 if (!impl_->getAccessToken(connection.get())) {
217 impl_->last_error_ = "Failed to get access token";
218 return "";
219 }
220
221 impl_->connections_[connection_id] = std::move(connection);
222
223 if (impl_->config_.enable_logging) {
224 spdlog::info("Firebase connection established: {}", connection_id);
225 }
226
227 return connection_id;
228}

230bool FirebaseManager::disconnect(const std::string& connection_id) {
231 std::lock_guard<std::mutex> lock(impl_->mutex_);
232
233 auto it = impl_->connections_.find(connection_id);
234 if (it == impl_->connections_.end()) {
235 impl_->last_error_ = "Connection not found: " + connection_id;
236 return false;
237 }
238
239 if (it->second->curl_handle) {
240 curl_easy_cleanup(it->second->curl_handle);
241 }
242
243 impl_->connections_.erase(it);
244
245 if (impl_->config_.enable_logging) {
246 spdlog::info("Firebase connection closed: {}", connection_id);
247 }
248
249 return true;
250}
