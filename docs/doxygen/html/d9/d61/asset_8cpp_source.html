50Asset AssetManager::createAsset(

62 throw std::runtime_error("Name, symbol, owner, and total supply are required");

72 asset.status = "active";
73
74 // Generate asset ID
75 asset.id = generateAssetId(asset);
76
77 // Initialize balance for owner

169bool AssetManager::transferAsset(
170 const std::string& assetId,
171 const std::string& from,
172 const std::string& to,
173 const std::string& amount) {
174
175 if (!initialized_) {
176 throw std::runtime_error("AssetManager not initialized");
177 }
178
179 if (assetId.empty() || from.empty() || to.empty() || amount.empty()) {
180 throw std::runtime_error("Asset ID, from, to, and amount are required");
181 }
182
183 if (assets_.find(assetId) == assets_.end()) {
184 return false;
185 }
186
187 auto& assetBalances = balances_[assetId];
188 if (assetBalances.find(from) == assetBalances.end()) {
189 return false;
190 }
191
192 // Check if sender has enough balance
193 if (assetBalances[from] < amount) {
194 return false;
195 }
196
197 // Update balances
198 assetBalances[from] = std::to_string(std::stod(assetBalances[from]) - std::stod(amount));
199 assetBalances[to] = std::to_string(std::stod(assetBalances[to]) + std::stod(amount));
200
201 return true;
202}

235std::string AssetManager::generateAssetId(const Asset& asset) {
236 std::stringstream ss;
237 ss << asset.name << asset.symbol << asset.owner
238 << asset.totalSupply << asset.createdAt.time_since_epoch().count();
239
240 // Generate SHA-256 hash
241 unsigned char hash[SHA256_DIGEST_LENGTH];
242 SHA256_CTX sha256;
243 SHA256_Init(&sha256);
244 SHA256_Update(&sha256, ss.str().c_str(), ss.str().size());
245 SHA256_Final(hash, &sha256);
246
247 // Convert to hex string
248 std::stringstream hexStream;
249 for (int i = 0; i < SHA256_DIGEST_LENGTH; i++) {
250 hexStream << std::hex << std::setw(2) << std::setfill('0') << static_cast<int>(hash[i]);
251 }
252
253 return hexStream.str();
254}
