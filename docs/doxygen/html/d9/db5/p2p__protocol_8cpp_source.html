35std::vector<uint8_t> doubleSHA256(const std::vector<uint8_t>& data) {
36 std::vector<uint8_t> hash1(SHA256_DIGEST_LENGTH);
37 std::vector<uint8_t> hash2(SHA256_DIGEST_LENGTH);
38
39 SHA256_CTX sha256;
40 SHA256_Init(&sha256);
41 SHA256_Update(&sha256, data.data(), data.size());
42 SHA256_Final(hash1.data(), &sha256);
43
44 SHA256_Init(&sha256);
45 SHA256_Update(&sha256, hash1.data(), hash1.size());
46 SHA256_Final(hash2.data(), &sha256);
47
48 return hash2;
49}

139std::vector<uint8_t> serializeMessage(const P2PMessage& message) {
140 std::vector<uint8_t> result;
141 result.reserve(sizeof(P2PMessageHeader) + message.payload.size());
142
143 // Serialize header
144 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&message.header.magic),
145 reinterpret_cast<const uint8_t*>(&message.header.magic) + sizeof(message.header.magic));
146 result.insert(result.end(), message.header.command,
147 message.header.command + sizeof(message.header.command));
148 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&message.header.length),
149 reinterpret_cast<const uint8_t*>(&message.header.length) + sizeof(message.header.length));
150 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&message.header.checksum),
151 reinterpret_cast<const uint8_t*>(&message.header.checksum) + sizeof(message.header.checksum));
152
153 // Add payload
154 result.insert(result.end(), message.payload.begin(), message.payload.end());
155
156 return result;
157}

159P2PMessage deserializeMessage(const std::vector<uint8_t>& data) {
160 if (data.size() < sizeof(P2PMessageHeader)) {
161 throw std::runtime_error("Message too short");
162 }
163
164 P2PMessage message;
165 size_t offset = 0;
166
167 // Deserialize header
168 std::memcpy(&message.header.magic, &data[offset], sizeof(message.header.magic));
169 offset += sizeof(message.header.magic);
170
171 std::memcpy(message.header.command, &data[offset], sizeof(message.header.command));
172 offset += sizeof(message.header.command);
173
174 std::memcpy(&message.header.length, &data[offset], sizeof(message.header.length));
175 offset += sizeof(message.header.length);
176
177 std::memcpy(&message.header.checksum, &data[offset], sizeof(message.header.checksum));
178 offset += sizeof(message.header.checksum);
179
180 // Verify magic number
181 if (message.header.magic != 0x52415645) {
182 throw std::runtime_error("Invalid magic number");
183 }
184
185 // Get payload
186 if (data.size() < offset + message.header.length) {
187 throw std::runtime_error("Message payload too short");
188 }
189
190 message.payload.assign(data.begin() + offset, data.begin() + offset + message.header.length);
191
192 // Verify checksum
193 uint32_t calculated_checksum = calculateChecksum(message.payload);
194 if (calculated_checksum != message.header.checksum) {
195 throw std::runtime_error("Invalid checksum");
196 }
197
198 return message;
199}

201std::vector<uint8_t> serializeVersionMessage(const VersionMessage& msg) {
202 std::vector<uint8_t> result;
203 result.reserve(100); // Approximate size
204
205 // Version
206 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.version),
207 reinterpret_cast<const uint8_t*>(&msg.version) + sizeof(msg.version));
208
209 // Services
210 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.services),
211 reinterpret_cast<const uint8_t*>(&msg.services) + sizeof(msg.services));
212
213 // Timestamp
214 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.timestamp),
215 reinterpret_cast<const uint8_t*>(&msg.timestamp) + sizeof(msg.timestamp));
216
217 // Address receiving
218 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.addr_recv_services),
219 reinterpret_cast<const uint8_t*>(&msg.addr_recv_services) + sizeof(msg.addr_recv_services));
220 // IP address (16 bytes)
221 std::vector<uint8_t> ip_bytes(16, 0);
222 if (!msg.addr_recv_ip.empty()) {
223 // Convert IP string to bytes (simplified)
224 std::string ip = msg.addr_recv_ip;
225 std::replace(ip.begin(), ip.end(), '.', ' ');
226 std::istringstream iss(ip);
227 int a, b, c, d;
228 iss >> a >> b >> c >> d;
229 ip_bytes[10] = 0xff;
230 ip_bytes[11] = 0xff;
231 ip_bytes[12] = a;
232 ip_bytes[13] = b;
233 ip_bytes[14] = c;
234 ip_bytes[15] = d;
235 }
236 result.insert(result.end(), ip_bytes.begin(), ip_bytes.end());
237
238 // Port
239 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.addr_recv_port),
240 reinterpret_cast<const uint8_t*>(&msg.addr_recv_port) + sizeof(msg.addr_recv_port));
241
242 // Address from
243 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.addr_from_services),
244 reinterpret_cast<const uint8_t*>(&msg.addr_from_services) + sizeof(msg.addr_from_services));
245 // IP address (16 bytes)
246 ip_bytes.assign(16, 0);
247 if (!msg.addr_from_ip.empty()) {
248 // Convert IP string to bytes (simplified)
249 std::string ip = msg.addr_from_ip;
250 std::replace(ip.begin(), ip.end(), '.', ' ');
251 std::istringstream iss(ip);
252 int a, b, c, d;
253 iss >> a >> b >> c >> d;
254 ip_bytes[10] = 0xff;
255 ip_bytes[11] = 0xff;
256 ip_bytes[12] = a;
257 ip_bytes[13] = b;
258 ip_bytes[14] = c;
259 ip_bytes[15] = d;
260 }
261 result.insert(result.end(), ip_bytes.begin(), ip_bytes.end());
262
263 // Port
264 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.addr_from_port),
265 reinterpret_cast<const uint8_t*>(&msg.addr_from_port) + sizeof(msg.addr_from_port));
266
267 // Nonce
268 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.nonce),
269 reinterpret_cast<const uint8_t*>(&msg.nonce) + sizeof(msg.nonce));
270
271 // User agent
272 uint8_t user_agent_size = static_cast<uint8_t>(msg.user_agent.size());
273 result.push_back(user_agent_size);
274 result.insert(result.end(), msg.user_agent.begin(), msg.user_agent.end());
275
276 // Start height
277 result.insert(result.end(), reinterpret_cast<const uint8_t*>(&msg.start_height),
278 reinterpret_cast<const uint8_t*>(&msg.start_height) + sizeof(msg.start_height));
279
280 // Relay
281 result.push_back(msg.relay ? 1 : 0);
282
283 return result;
284}
