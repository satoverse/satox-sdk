78// P2P message header structure
79struct P2PMessageHeader {
80 uint32_t magic; // Network magic number (0x52415645 for Satoxcoin)
81 char command[12]; // Command name
82 uint32_t length; // Payload length
83 uint32_t checksum; // First 4 bytes of double SHA256 of payload
84
85 P2PMessageHeader() : magic(0x52415645), length(0), checksum(0) {
86 std::fill(command, command + 12, 0);
87 }
88};
