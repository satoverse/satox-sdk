155nlohmann::json RPCManager::handleRequest(const nlohmann::json& request) {
156 try {
157 if (!request.contains("method")) {
158 return createErrorResponse(-32600, "Invalid request: missing method");
159 }
160
161 const std::string& method = request["method"];
162 const nlohmann::json& params = request.value("params", nlohmann::json::array());
163
164 auto it = method_handlers_.find(method);
165 if (it == method_handlers_.end()) {
166 return createErrorResponse(-32601, "Method not found: " + method);
167 }
168
169 return it->second(params);
170 } catch (const std::exception& e) {
171 spdlog::error("RPC request handling error: {}", e.what());
172 return createErrorResponse(-32603, "Internal error: " + std::string(e.what()));
173 }
174}

195bool RPCManager::validateParams(const nlohmann::json& params,
196 const std::vector<std::string>& required) {
197 if (!params.is_array()) {
198 return false;
199 }
200
201 for (const auto& param : required) {
202 if (!params.contains(param)) {
203 return false;
204 }
205 }
206
207 return true;
