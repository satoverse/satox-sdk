154template<typename T>
155class LRUCache {
156public:
157 explicit LRUCache(size_t capacity) : capacity_(capacity) {}
158
159 void put(const std::string& key, const T& value) {
160 std::lock_guard<std::mutex> lock(mutex_);
161 auto it = cache_.find(key);
162 if (it != cache_.end()) {
163 list_.erase(it->second.second);
164 } else if (cache_.size() >= capacity_) {
165 cache_.erase(list_.back());
166 list_.pop_back();
167 }
168 list_.push_front(key);
169 cache_[key] = {value, list_.begin()};
170 }
171
172 std::optional<T> get(const std::string& key) {
173 std::lock_guard<std::mutex> lock(mutex_);
174 auto it = cache_.find(key);
175 if (it == cache_.end()) {
176 return std::nullopt;
177 }
178 list_.splice(list_.begin(), list_, it->second.second);
179 return it->second.first;
180 }
181
182 std::optional<T> get(const std::string& key) const {
183 std::lock_guard<std::mutex> lock(mutex_);
184 auto it = cache_.find(key);
185 if (it == cache_.end()) {
186 return std::nullopt;
187 }
188 return it->second.first;
189 }
190
191 void clear() {
192 std::lock_guard<std::mutex> lock(mutex_);
193 cache_.clear();
194 list_.clear();
195 }
196
197private:
198 size_t capacity_;
199 std::list<std::string> list_;
200 std::unordered_map<std::string, std::pair<T, std::list<std::string>::iterator>> cache_;
201 mutable std::mutex mutex_;
202};
