53 std::string calculateCID(const std::vector<uint8_t>& data) {
54 unsigned char hash[SHA256_DIGEST_LENGTH];
55 SHA256_CTX sha256;
56 SHA256_Init(&sha256);
57 SHA256_Update(&sha256, data.data(), data.size());
58 SHA256_Final(hash, &sha256);
59
60 std::stringstream ss;
61 ss << "Qm";
62 for (int i = 0; i < SHA256_DIGEST_LENGTH; i++) {

720std::string IPFSManager::generateNodeId() {
721 static std::random_device rd;
722 static std::mt19937 gen(rd());
723 static std::uniform_int_distribution<> dis(0, 15);
724 static const char* hex = "0123456789abcdef";
725
726 std::string id;
727 id.reserve(46);
728 id += "Qm";
729
730 for (int i = 0; i < 44; ++i) {
731 id += hex[dis(gen)];
732 }

737std::string IPFSManager::calculateContentHash(const std::vector<uint8_t>& data) {
738 // Implement proper IPFS CID calculation using SHA-256
739 try {
740 if (data.empty()) {
741 return "";
742 }
743
744 // Calculate SHA-256 hash
745 unsigned char hash[SHA256_DIGEST_LENGTH];
746 SHA256_CTX sha256;
747 SHA256_Init(&sha256);
748 SHA256_Update(&sha256, data.data(), data.size());
749 SHA256_Final(hash, &sha256);
750
751 // Convert to base58 (IPFS CID format)
752 // For simplicity, we'll use a basic base58 encoding
753 // In a real implementation, you'd use a proper base58 library
754 std::string base58 = "123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz";
755 std::string result = "Qm"; // IPFS CIDv0 prefix
756
757 // Convert hash to base58
758 std::vector<uint8_t> hashVec(hash, hash + SHA256_DIGEST_LENGTH);
759 std::string base58Result = encodeBase58(hashVec);
760
761 // Ensure it's 44 characters (IPFS CIDv0 length)
762 if (base58Result.length() > 44) {
763 base58Result = base58Result.substr(0, 44);
764 } else while (base58Result.length() < 44) {
765 base58Result = "1" + base58Result;
766 }
767
768 return result + base58Result;
769 } catch (const std::exception& e) {
770 return "";
771 }
772}

774// Helper method for base58 encoding
775std::string IPFSManager::encodeBase58(const std::vector<uint8_t>& data) {
776 const std::string base58 = "123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz";
777 std::string result;
778
779 // Convert to base58
780 std::vector<uint8_t> digits((data.size() * 138 / 100) + 1);
781 size_t digitslen = 1;
782
783 for (size_t i = 0; i < data.size(); i++) {
784 uint32_t carry = data[i];
785 for (size_t j = 0; j < digitslen; j++) {
786 carry += static_cast<uint32_t>(digits[j]) << 8;
787 digits[j] = carry % 58;
788 carry /= 58;
789 }
790 while (carry > 0) {
791 digits[digitslen++] = carry % 58;
792 carry /= 58;
793 }
794 }
795
796 // Convert to string
797 for (size_t i = 0; i < digitslen; i++) {
798 result += base58[digits[digitslen - 1 - i]];
799 }
800
801 // Remove leading zeros
802 while (result.length() > 0 && result[0] == '1') {
803 result = result.substr(1);
804 }
805
806 return result;
807}
