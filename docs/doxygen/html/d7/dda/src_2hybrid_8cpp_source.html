79 bool hybridEncrypt(const std::vector<uint8_t>& publicKey,
80 const std::vector<uint8_t>& message,
81 std::vector<uint8_t>& ciphertext) {
82 try {
83 // Split public key
84 auto [kemPublicKey, dsaPublicKey] = splitKey(publicKey);
85
86 // Generate ephemeral key pair
87 MLKEM kem(kemLevel);
88 std::vector<uint8_t> ephemeralPublicKey, ephemeralPrivateKey;
89 if (!kem.generateKeyPair(ephemeralPublicKey, ephemeralPrivateKey)) {
90 throw std::runtime_error("Failed to generate ephemeral key pair: " +
91 kem.getLastError());
92 }
93
94 // Encapsulate shared secret
95 std::vector<uint8_t> kemCiphertext, sharedSecret;
96 if (!kem.encapsulate(kemPublicKey, kemCiphertext, sharedSecret)) {
97 throw std::runtime_error("Failed to encapsulate: " +
98 kem.getLastError());
99 }
100
101 // Encrypt message with shared secret
102 std::vector<uint8_t> encryptedMessage;
103 if (!encryptMessage(message, sharedSecret, encryptedMessage)) {
104 throw std::runtime_error("Failed to encrypt message");
105 }
106
107 // Sign ephemeral public key
108 MLDSA dsa(dsaLevel);
109 std::vector<uint8_t> signature;
110 if (!dsa.sign(dsaPublicKey, ephemeralPublicKey, signature)) {
111 throw std::runtime_error("Failed to sign: " +
112 dsa.getLastError());
113 }
114
115 // Combine components
116 ciphertext = combineComponents(kemCiphertext, encryptedMessage,
117 ephemeralPublicKey, signature);
118
119 return true;
120 } catch (const std::exception& e) {
121 lastError = e.what();
122 return false;
123 }
124 }

126 bool hybridDecrypt(const std::vector<uint8_t>& privateKey,
127 const std::vector<uint8_t>& ciphertext,
128 std::vector<uint8_t>& message) {
129 try {
130 // Split private key
131 auto [kemPrivateKey, dsaPrivateKey] = splitKey(privateKey);
132
133 // Split ciphertext components
134 auto [kemCiphertext, encryptedMessage, ephemeralPublicKey, signature] =
135 splitComponents(ciphertext);
136
137 // Verify signature
138 MLDSA dsa(dsaLevel);
139 if (!dsa.verify(dsaPrivateKey, ephemeralPublicKey, signature)) {
140 throw std::runtime_error("Signature verification failed: " +
141 dsa.getLastError());
142 }
143
144 // Decapsulate shared secret
145 MLKEM kem(kemLevel);
146 std::vector<uint8_t> sharedSecret;
147 if (!kem.decapsulate(kemPrivateKey, kemCiphertext, sharedSecret)) {
148 throw std::runtime_error("Failed to decapsulate: " +
149 kem.getLastError());
150 }
151
152 // Decrypt message
153 if (!decryptMessage(encryptedMessage, sharedSecret, message)) {
154 throw std::runtime_error("Failed to decrypt message");
155 }
156
157 return true;
158 } catch (const std::exception& e) {
159 lastError = e.what();
160 return false;
161 }

269 std::pair<std::vector<uint8_t>, std::vector<uint8_t>>

271 MLKEM kem(kemLevel);

306 std::tuple<std::vector<uint8_t>, std::vector<uint8_t>,
307 std::vector<uint8_t>, std::vector<uint8_t>>
308 splitComponents(const std::vector<uint8_t>& combined) {
309 MLKEM kem(kemLevel);
310 MLDSA dsa(dsaLevel);
311
312 size_t kemCiphertextSize = kem.getCiphertextSize();
313 size_t dsaSignatureSize = dsa.getSignatureSize();
314 size_t kemPublicKeySize = kem.getPublicKeySize();
315
316 auto it = combined.begin();
317 std::vector<uint8_t> kemCiphertext(it, it + kemCiphertextSize);
318 it += kemCiphertextSize;
319
320 std::vector<uint8_t> encryptedMessage(it, it + 32);
321 it += 32;
322
323 std::vector<uint8_t> ephemeralPublicKey(it, it + kemPublicKeySize);
324 it += kemPublicKeySize;
325
326 std::vector<uint8_t> signature(it, it + dsaSignatureSize);
327
328 return {kemCiphertext, encryptedMessage, ephemeralPublicKey, signature};
329 }

331 std::vector<uint8_t> combineComponents(
332 const std::vector<uint8_t>& kemCiphertext,
333 const std::vector<uint8_t>& encryptedMessage,
334 const std::vector<uint8_t>& ephemeralPublicKey,
335 const std::vector<uint8_t>& signature) {
336
337 std::vector<uint8_t> combined;
338 combined.reserve(kemCiphertext.size() + encryptedMessage.size() +
339 ephemeralPublicKey.size() + signature.size());

341 combined.insert(combined.end(), kemCiphertext.begin(), kemCiphertext.end());
342 combined.insert(combined.end(), encryptedMessage.begin(), encryptedMessage.end());
343 combined.insert(combined.end(), ephemeralPublicKey.begin(), ephemeralPublicKey.end());
344 combined.insert(combined.end(), signature.begin(), signature.end());
345
