352uint64_t BlockchainManager::getDifficulty() const {
353 std::lock_guard<std::mutex> lock(mutex_);
354
355 if (blocks_.empty()) {
356 return 0;
357 }
358
359 // Calculate difficulty based on the last 2016 blocks (similar to Bitcoin)
360 const uint64_t difficultyAdjustmentInterval = 2016;
361 const uint64_t targetTimeSpan = 14 * 24 * 60 * 60; // 14 days in seconds
362
363 uint64_t currentHeight = getCurrentHeight();
364 if (currentHeight < difficultyAdjustmentInterval) {
365 return 1; // Default difficulty for early blocks
366 }
367
368 // Get the block from 2016 blocks ago
369 uint64_t oldHeight = currentHeight - difficultyAdjustmentInterval;
370 auto oldBlockIt = std::find_if(blocks_.begin(), blocks_.end(),
371 [oldHeight](const auto& pair) { return pair.second.height == oldHeight; });
372
373 if (oldBlockIt == blocks_.end()) {
374 return 1; // Fallback difficulty
375 }
376
377 // Get current block
378 auto currentBlockIt = std::find_if(blocks_.begin(), blocks_.end(),
379 [currentHeight](const auto& pair) { return pair.second.height == currentHeight; });
380
381 if (currentBlockIt == blocks_.end()) {
382 return 1; // Fallback difficulty
383 }
384
385 // Calculate time span
386 auto timeSpan = currentBlockIt->second.timestamp - oldBlockIt->second.timestamp;
387 if (timeSpan == 0) {
388 timeSpan = 1; // Avoid division by zero
389 }
390
391 // Calculate difficulty adjustment based on bits field
392 // The bits field encodes the target difficulty
393 uint32_t oldBits = oldBlockIt->second.bits;
394 if (oldBits == 0) {
395 oldBits = 0x1d00ffff; // Default difficulty target
396 }
397
398 // Convert bits to target
399 uint64_t oldTarget = (oldBits & 0xffffff) << (8 * ((oldBits >> 24) - 3));
400
401 // Calculate new target based on time span
402 uint64_t newTarget = (oldTarget * targetTimeSpan) / timeSpan;
403
404 // Limit target adjustment to 4x increase or 1/4 decrease
405 if (newTarget > oldTarget * 4) {
406 newTarget = oldTarget * 4;
407 } else if (newTarget < oldTarget / 4) {
408 newTarget = oldTarget / 4;
409 }
410
411 // Convert target back to difficulty (difficulty = target_max / target)
412 const uint64_t targetMax = 0x00000000ffff0000ULL;
413 uint64_t difficulty = targetMax / newTarget;
414
415 return difficulty;
416}
