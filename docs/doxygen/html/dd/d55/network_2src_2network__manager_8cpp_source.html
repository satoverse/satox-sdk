360bool NetworkManager::Impl::initialize(const NetworkConfig& config) {
361 std::unique_lock<std::shared_mutex> lock(mutex_);
362
363 if (initialized_) {
364 logError("NetworkManager already initialized");
365 return false;
366 }
367
368 if (!validateConfig(config)) {
369 logError("Invalid configuration");
370 return false;
371 }
372
373 try {
374 config_ = config;
375
376 // Initialize SSL context if enabled
377 if (config.enable_ssl) {
378 ssl_context_ = std::make_shared<boost::asio::ssl::context>(boost::asio::ssl::context::sslv23);
379 ssl_context_->set_verify_mode(boost::asio::ssl::verify_peer);
380 ssl_context_->set_default_verify_paths();
381 }
382
383 // Initialize Let's Encrypt manager if configured
384 if (!config.letsencrypt_config.domain.empty()) {
385 letsencrypt_manager_ = std::make_unique<LetsEncryptManager>();
386 if (!letsencrypt_manager_->initialize(config.letsencrypt_config)) {
387 logError("Failed to initialize Let's Encrypt manager");
388 return false;
389 }
390 }
391
392 initialized_ = true;
393 logInfo("NetworkManager initialized successfully");
394 return true;
395 } catch (const std::exception& e) {
396 logError("Initialization failed: " + std::string(e.what()));
397 return false;
398 }
399}

401bool NetworkManager::Impl::shutdown() {
402 std::unique_lock<std::shared_mutex> lock(mutex_);
403
404 if (!initialized_) {
405 return true;
406 }
407
408 try {
409 shutdown_requested_ = true;
410
411 // Stop P2P and RPC
412 if (p2p_running_) {
413 stopP2P();
414 }
415 if (rpc_running_) {
416 stopRPC();
417 }
418
419 // Wait for threads to finish
420 if (p2p_thread_ && p2p_thread_->joinable()) {
421 p2p_thread_->join();
422 }
423 if (rpc_thread_ && rpc_thread_->joinable()) {
424 rpc_thread_->join();
425 }
426
427 // Clear callbacks
428 message_callbacks_.clear();
429 rpc_handlers_.clear();
430 peers_.clear();
431
432 initialized_ = false;
433 logInfo("NetworkManager shutdown completed");
434 return true;
435 } catch (const std::exception& e) {
436 logError("Shutdown failed: " + std::string(e.what()));
437 return false;
438 }

500bool NetworkManager::Impl::connectToPeer(const std::string& address, uint16_t port) {
501 std::unique_lock<std::shared_mutex> lock(mutex_);
502
503 if (!initialized_) {
504 logError("NetworkManager not initialized");
505 return false;
506 }
507
508 try {
509 Peer peer;
510 peer.address = address;
511 peer.port = port;
512 peer.last_seen = std::chrono::system_clock::now().time_since_epoch().count();
513 peer.inbound = false;
514 peer.whitelisted = false;
515 peer.bytes_sent = 0;
516 peer.bytes_received = 0;
517
518 peers_[address] = peer;
519
520 if (peer_callback_) {
521 peer_callback_(address, true);
522 }
523
524 logInfo("Connected to peer: " + address + ":" + std::to_string(port));
525 return true;
526 } catch (const std::exception& e) {
527 logError("Failed to connect to peer: " + std::string(e.what()));
528 return false;
529 }
530}
