50bool Kawpow::computeHash(const std::vector<uint8_t>& header,
51 uint64_t nonce,
52 std::vector<uint8_t>& hash,
53 std::vector<uint8_t>& mix_hash) {
54 if (header.size() != KAWPOW_HEADER_BYTES) {
55 return false;
56 }
57
58 // Initialize cache and dataset if needed
59 uint64_t block_number = 0; // TODO: Get from header
60 if (!initializeCache(block_number) || !initializeDataset(block_number)) {
61 return false;
62 }
63
64 // Calculate header hash
65 std::vector<uint8_t> header_hash(KAWPOW_HASH_BYTES);
66 hashHeader(header, header_hash);
67
68 // Calculate nonce hash
69 std::vector<uint8_t> nonce_hash(KAWPOW_HASH_BYTES);
70 hashNonce(nonce, nonce_hash);
71
72 // Combine header and nonce hashes
73 std::vector<uint8_t> seed(KAWPOW_HASH_BYTES);
74 for (size_t i = 0; i < KAWPOW_HASH_BYTES; ++i) {
75 seed[i] = header_hash[i] ^ nonce_hash[i];
76 }
77
78 // Generate mix hash
79 mix_hash.resize(KAWPOW_MIX_BYTES);
80 mixHash(seed, mix_hash);
81
82 // Calculate final hash
83 hash.resize(KAWPOW_HASH_BYTES);
84 SHA256_CTX sha256;
85 SHA256_Init(&sha256);
86 SHA256_Update(&sha256, mix_hash.data(), mix_hash.size());
87 SHA256_Final(hash.data(), &sha256);
88
89 return true;
90}

92bool Kawpow::verifyHash(const std::vector<uint8_t>& header,
93 uint64_t nonce,
94 const std::vector<uint8_t>& mix_hash,
95 const std::vector<uint8_t>& target) {
96 if (header.size() != KAWPOW_HEADER_BYTES || mix_hash.size() != KAWPOW_MIX_BYTES) {
97 return false;
98 }
99
100 std::vector<uint8_t> hash;
101 std::vector<uint8_t> computed_mix_hash;
102 if (!computeHash(header, nonce, hash, computed_mix_hash)) {
103 return false;
104 }
105
106 // Verify mix hash
107 if (computed_mix_hash != mix_hash) {
108 return false;
109 }
110
111 // Verify target
112 return std::lexicographical_compare(hash.begin(), hash.end(),
113 target.begin(), target.end());
114}

165uint64_t Kawpow::getCacheSize(uint64_t block_number) const {
166 uint64_t epoch = getEpoch(block_number);
167 uint64_t size = KAWPOW_CACHE_BYTES_INIT;
168 for (uint64_t i = 0; i < epoch; ++i) {
169 size = size * 3 / 2;
170 size = (size + KAWPOW_CACHE_BYTES_GROWTH - 1) & ~(KAWPOW_CACHE_BYTES_GROWTH - 1);

179bool Kawpow::generateCache(uint64_t block_number) {
180 if (cache_.empty()) {
181 return false;
182 }
183
184 // Initialize Keccak sponge
185 KeccakSponge sponge;
186 KeccakSponge_Initialize(&sponge, KECCAK_SPONGE_BITRATE, KECCAK_SPONGE_CAPACITY);
187
188 // Generate seed
189 std::vector<uint8_t> seed(32);
190 for (size_t i = 0; i < 32; ++i) {
191 seed[i] = static_cast<uint8_t>(block_number >> (i * 8));
192 }
193
194 // Generate cache using Keccak-512
195 for (size_t i = 0; i < cache_.size(); i += 64) {
196 KeccakSponge_Absorb(&sponge, seed.data(), seed.size());
197 KeccakSponge_Squeeze(&sponge, cache_.data() + i, std::min(64UL, cache_.size() - i));
198 }
199
200 // Mix the cache
201 for (size_t i = 0; i < KAWPOW_CACHE_ROUNDS; ++i) {
202 for (size_t j = 0; j < cache_.size(); j += 64) {
203 uint64_t v = 0;
204 for (size_t k = 0; k < 8; ++k) {
205 v ^= *reinterpret_cast<uint64_t*>(cache_.data() + ((j + k * 8) % cache_.size()));
206 }
207 for (size_t k = 0; k < 8; ++k) {
208 *reinterpret_cast<uint64_t*>(cache_.data() + ((j + k * 8) % cache_.size())) ^= v;
209 }
210 }
211 }
212
213 return true;
214}

216bool Kawpow::generateDataset(uint64_t block_number) {
217 if (dataset_.empty() || cache_.empty()) {
218 return false;
219 }
220
221 // Generate dataset using cache
222 for (size_t i = 0; i < dataset_.size(); i += 64) {
223 uint64_t index = i / 64;
224 std::vector<uint8_t> mix(64);
225
226 // Initialize mix with cache data
227 size_t cache_index = (index % (cache_.size() / 64)) * 64;
228 std::memcpy(mix.data(), cache_.data() + cache_index, 64);
229
230 // Mix with parent nodes
231 for (size_t j = 0; j < KAWPOW_DATASET_PARENTS; ++j) {
232 uint64_t parent_index = (index * KAWPOW_DATASET_PARENTS + j) % (cache_.size() / 64);
233 size_t parent_offset = parent_index * 64;
234
235 for (size_t k = 0; k < 64; ++k) {
236 mix[k] ^= cache_[parent_offset + k];
237 }
238 }
239
240 // Store mixed data
241 std::memcpy(dataset_.data() + i, mix.data(), 64);
242 }
243
244 return true;
245}

261void Kawpow::mixHash(const std::vector<uint8_t>& seed, std::vector<uint8_t>& mix_hash) {
262 if (dataset_.empty()) {
263 std::fill(mix_hash.begin(), mix_hash.end(), 0);
264 return;
265 }
266
267 // Initialize mix with seed
268 std::vector<uint8_t> mix(64);
269 std::memcpy(mix.data(), seed.data(), std::min(seed.size(), size_t(64)));
270
271 // Mix with dataset
272 for (size_t i = 0; i < KAWPOW_ACCESSES; ++i) {
273 uint64_t index = 0;
274 for (size_t j = 0; j < 8; ++j) {
275 index = (index << 8) | mix[j];
276 }
277 index %= dataset_.size() / 64;
278
279 for (size_t j = 0; j < 64; ++j) {
280 mix[j] ^= dataset_[index * 64 + j];
281 }
282 }
283
284 // Store final mix
285 std::memcpy(mix_hash.data(), mix.data(), std::min(mix.size(), mix_hash.size()));
286}
