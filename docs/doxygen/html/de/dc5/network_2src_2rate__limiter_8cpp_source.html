152bool RateLimiter::checkLimit(const std::string& method, const std::string& client_id) {
153 std::lock_guard<std::mutex> lock(mutex_);
154 if (!running_) {
155 throw std::runtime_error("RateLimiter not running");
156 }
157
158 auto now = std::chrono::steady_clock::now();
159 auto& client_counts = request_counts_[client_id];
160 auto& count = client_counts[method];
161
162 // Get the appropriate limit configuration
163 const LimitConfig* limit = nullptr;
164 if (auto it = method_limits_.find(method); it != method_limits_.end()) {
165 limit = &it->second;
166 } else if (auto it = client_limits_.find(client_id); it != client_limits_.end()) {
167 limit = &it->second;
168 } else {
169 limit = &default_limit_;
170 }
171
172 // Check if the window has expired
173 if (now - count.window_start > limit->window) {
174 count.count = 0;
175 count.window_start = now;
176 }
177
178 // Check if the limit has been exceeded
179 return count.count < limit->max_requests;
180}
