63 bool initialize(const Config& config) {
64 std::lock_guard<std::mutex> lock(mutex);
65
66 if (is_initialized) {

70
71 // Validate configuration

73 last_error = "Invalid configuration: missing required fields";

78 try {

97 current_cert.key_path = config.cert_path + "/privkey.pem";
98 current_cert.chain_path = config.cert_path + "/chain.pem";
99 current_cert.domain = config.domain;

102 // If certificate is expired or near expiry, trigger renewal

143 bool renewCertificate(const std::string& domain = "") {
144 std::lock_guard<std::mutex> lock(mutex);
145
146 if (!is_initialized) {
147 last_error = "Not initialized";
148 return false;
149 }
150
151 // If domain is provided, use it instead of config domain
152 std::string target_domain = domain.empty() ? config.domain : domain;
153
154 try {
155 spdlog::info("Starting certificate renewal for domain: {}", target_domain);
156
157 // Generate new private key
158 std::string key_path = config.cert_path + "/privkey.pem";
159 if (!generatePrivateKey(key_path)) {
160 spdlog::error("Failed to generate private key: {}", last_error);
161 return false;
162 }
163
164 // Generate CSR
165 std::string csr_path = config.cert_path + "/domain.csr";
166 if (!generateCSR(csr_path, key_path)) {
167 spdlog::error("Failed to generate CSR: {}", last_error);
168 return false;
169 }
170
171 // Request certificate from Let's Encrypt
172 std::string cert_path = config.cert_path + "/cert.pem";
173 std::string chain_path = config.cert_path + "/chain.pem";
174 if (!requestCertificate(csr_path, cert_path, chain_path)) {
175 spdlog::error("Failed to request certificate: {}", last_error);
176 return false;
177 }
178
179 // Verify the new certificate
180 if (!verifyCertificate(cert_path, key_path)) {
181 spdlog::error("Failed to verify new certificate: {}", last_error);
182 return false;
183 }
184
185 // Backup old certificate if it exists
186 if (std::filesystem::exists(current_cert.cert_path)) {
187 std::string backup_path = current_cert.cert_path + ".bak";
188 std::filesystem::rename(current_cert.cert_path, backup_path);
189 }
190
191 // Update current certificate
192 current_cert.cert_path = cert_path;
193 current_cert.key_path = key_path;
194 current_cert.chain_path = chain_path;
195 current_cert.domain = target_domain;
196 current_cert.expiry = getCertificateExpiry(cert_path);
197
198 spdlog::info("Certificate renewed successfully for domain: {}", target_domain);
199 spdlog::info("Certificate expires at: {}", std::chrono::system_clock::to_time_t(current_cert.expiry));
200
201 // Notify callback
202 if (update_callback) {
203 update_callback(current_cert);
204 }
205
206 return true;
207 } catch (const std::exception& e) {
208 last_error = std::string("Certificate renewal failed: ") + e.what();
209 spdlog::error("Certificate renewal failed: {}", last_error);
210 return false;

230 bool generatePrivateKey(const std::string& path) {
231 try {
232 // Generate RSA key
233 RSA* rsa = RSA_generate_key(2048, RSA_F4, nullptr, nullptr);
234 if (!rsa) {
235 last_error = "Failed to generate RSA key";
236 return false;
237 }
238
239 // Write private key to file
240 FILE* fp = fopen(path.c_str(), "w");
241 if (!fp) {
242 RSA_free(rsa);
243 last_error = "Failed to open private key file for writing";
244 return false;
245 }
246
247 if (!PEM_write_RSAPrivateKey(fp, rsa, nullptr, nullptr, 0, nullptr, nullptr)) {
248 fclose(fp);
249 RSA_free(rsa);
250 last_error = "Failed to write private key";
251 return false;
252 }
253
254 fclose(fp);
255 RSA_free(rsa);
256 return true;
257 } catch (const std::exception& e) {
258 last_error = std::string("Error generating private key: ") + e.what();
259 return false;

263 bool generateCSR(const std::string& path, const std::string& key_path) {
264 try {
265 // Read private key
266 FILE* fp = fopen(key_path.c_str(), "r");
267 if (!fp) {
268 last_error = "Failed to open private key file";
269 return false;
270 }
271
272 RSA* rsa = PEM_read_RSAPrivateKey(fp, nullptr, nullptr, nullptr);
273 fclose(fp);
274
275 if (!rsa) {
276 last_error = "Failed to read private key";
277 return false;
278 }
279
280 // Create CSR
281 X509_REQ* req = X509_REQ_new();
282 if (!req) {
283 RSA_free(rsa);
284 last_error = "Failed to create CSR";
285 return false;
286 }
287
288 // Set subject
289 X509_NAME* name = X509_REQ_get_subject_name(req);
290 X509_NAME_add_entry_by_txt(name, "CN", MBSTRING_ASC,
291 (unsigned char*)config.domain.c_str(), -1, -1, 0);
292
293 // Set public key
294 EVP_PKEY* pkey = EVP_PKEY_new();
295 EVP_PKEY_assign_RSA(pkey, rsa);
296 X509_REQ_set_pubkey(req, pkey);
297
298 // Sign CSR
299 if (!X509_REQ_sign(req, pkey, EVP_sha256())) {
300 X509_REQ_free(req);
301 EVP_PKEY_free(pkey);
302 last_error = "Failed to sign CSR";
303 return false;
304 }
305
306 // Write CSR to file
307 fp = fopen(path.c_str(), "w");
308 if (!fp) {
309 X509_REQ_free(req);
310 EVP_PKEY_free(pkey);
311 last_error = "Failed to open CSR file for writing";
312 return false;
313 }
314
315 PEM_write_X509_REQ(fp, req);
316 fclose(fp);
317 X509_REQ_free(req);
318 EVP_PKEY_free(pkey);
319
320 return true;
321 } catch (const std::exception& e) {
322 last_error = std::string("Error generating CSR: ") + e.what();
323 return false;
324 }

336 std::chrono::system_clock::time_point getCertificateExpiry(const std::string& cert_path) {
337 try {
338 FILE* fp = fopen(cert_path.c_str(), "r");
339 if (!fp) {
340 last_error = "Failed to open certificate file";
341 return std::chrono::system_clock::now();
342 }
343
344 X509* cert = PEM_read_X509(fp, nullptr, nullptr, nullptr);
345 fclose(fp);
346
347 if (!cert) {
348 last_error = "Failed to read certificate";
349 return std::chrono::system_clock::now();
350 }
351
352 ASN1_TIME* notAfter = X509_get_notAfter(cert);
353 if (!notAfter) {
354 X509_free(cert);
355 last_error = "Failed to get certificate expiry";
356 return std::chrono::system_clock::now();
357 }
358
359 // Convert ASN1_TIME to system_clock::time_point
360 struct tm tm;
361 ASN1_TIME_to_tm(notAfter, &tm);
362 auto time = std::mktime(&tm);
363 auto expiry = std::chrono::system_clock::from_time_t(time);
364
365 X509_free(cert);
366 return expiry;
367 } catch (const std::exception& e) {
368 last_error = std::string("Error getting certificate expiry: ") + e.what();
369 return std::chrono::system_clock::now();
370 }

404 return false;
405 }
406
407 // Verify certificate signature
408 bool valid = X509_verify(cert, pkey) == 1;
409

425 void startRenewalThread() {
426 if (renewal_thread_running) {
427 return;
428 }
429
430 renewal_thread_running = true;
431 renewal_thread = std::thread([this]() {
432 while (renewal_thread_running) {
433 try {
434 auto now = std::chrono::system_clock::now();
435 auto expiry = current_cert.expiry;
436 auto threshold = config.renewal_threshold;
437
438 // Check if certificate needs renewal
439 if (expiry - now < threshold) {
440 spdlog::info("Certificate approaching expiry, initiating renewal");
441 if (!renewCertificate()) {
442 spdlog::error("Automatic renewal failed: {}", last_error);
443 // If renewal failed, try again in 5 minutes
444 std::this_thread::sleep_for(std::chrono::minutes(5));
445 continue;
446 }
447 }
448
449 // Calculate next check time
450 auto next_check = std::min(
451 std::chrono::hours(1), // Check at least every hour
452 std::chrono::duration_cast<std::chrono::hours>(expiry - now - threshold) / 2 // Or halfway to renewal threshold
453 );
454
455 // Wait for next check or shutdown signal
456 std::unique_lock<std::mutex> lock(mutex);
457 renewal_cv.wait_for(lock, next_check, [this]() {
458 return !renewal_thread_running;
459 });
460 } catch (const std::exception& e) {
461 spdlog::error("Error in renewal thread: {}", e.what());
462 std::this_thread::sleep_for(std::chrono::minutes(5)); // Wait before retrying
463 }
464 }
465 });
466 }
