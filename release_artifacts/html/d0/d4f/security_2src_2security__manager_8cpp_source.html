166bool SecurityManager::performOperation(const std::string& operation, const nlohmann::json& params) {
167 std::lock_guard<std::mutex> lock(mutex_);
168
169 if (!validateState()) {
170 return false;
171 }
172
173 auto start_time = std::chrono::high_resolution_clock::now();
174 bool success = false;
175
176 try {
177 // Validate operation
178 if (!validateOperation(operation, params)) {
179 setLastError("Operation validation failed");
180 return false;
181 }
182
183 // Perform the operation
184 success = executeOperation(operation, params).success;
185
186 // Update statistics
187 auto end_time = std::chrono::high_resolution_clock::now();
188 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
189 updateStats(success, duration.count());
190
191 // Log operation
192 logOperation(operation, success, duration.count());
193
194 return success;
195 } catch (const std::exception& e) {
196 setLastError(std::string("Operation failed: ") + e.what());
197 return false;
198 }
199}

201SecurityResult SecurityManager::executeOperation(const std::string& operation, const nlohmann::json& params) {
202 SecurityResult result;
203 result.success = false;
204
205 try {
206 // Basic operation execution
207 if (operation == "validate_input") {
208 std::string input = params.value("input", "");
209 std::string type = params.value("type", "string");
210 result.success = validateInput(input, type);
211 } else if (operation == "check_rate_limit") {
212 std::string identifier = params.value("identifier", "");
213 std::string op = params.value("operation", "");
214 result.success = checkRateLimit(identifier, op);
215 } else {
216 result.error = "Unknown operation: " + operation;
217 return result;
218 }
219
220 result.duration = std::chrono::milliseconds(0);
221 return result;
222 } catch (const std::exception& e) {
223 result.error = std::string("Execution failed: ") + e.what();
224 return result;
225 }

517void SecurityManager::updateStats(bool success, double operationTime) {
518 if (!statsEnabled_) return;
519
520 stats_.totalOperations++;
521 if (success) {
522 stats_.successfulOperations++;
523 consecutiveFailures_ = 0;
524 } else {
525 stats_.failedOperations++;
526 }
527
528 stats_.lastOperation = std::chrono::system_clock::now();
529
530 // Update average operation time
531 if (stats_.totalOperations > 1) {
532 stats_.averageOperationTime =
533 (stats_.averageOperationTime * (stats_.totalOperations - 1) + operationTime) / stats_.totalOperations;
534 } else {
535 stats_.averageOperationTime = operationTime;
536 }
537}
