50 bool generateAndStoreKey(const std::string& keyId,
51 MLKEM::SecurityLevel kemLevel,
52 MLDSA::SecurityLevel dsaLevel,
53 const std::chrono::seconds& expirationTime) {
54 try {
55 std::lock_guard<std::mutex> lock(mutex);
56
57 // Check if key already exists
58 if (keys.find(keyId) != keys.end()) {
59 throw std::runtime_error("Key with ID " + keyId + " already exists");
60 }
61
62 // Generate new key pair
63 HybridCrypto crypto(kemLevel, dsaLevel);
64 std::vector<uint8_t> publicKey, privateKey;
65 if (!crypto.generateHybridKeyPair(publicKey, privateKey)) {
66 throw std::runtime_error("Failed to generate key pair: " +
67 crypto.getLastError());
68 }
69
70 // Create key metadata
71 KeyMetadata metadata;
72 metadata.keyId = keyId;
73 metadata.creationTime = std::chrono::system_clock::now();
74 metadata.expirationTime = metadata.creationTime + expirationTime;
75 metadata.algorithm = "ML-KEM-" + std::to_string(static_cast<int>(kemLevel)) +
76 "+ML-DSA-" + std::to_string(static_cast<int>(dsaLevel));
77 metadata.securityLevel = std::max(static_cast<int>(kemLevel),
78 static_cast<int>(dsaLevel));
79 metadata.isActive = true;
80
81 // Store key and metadata
82 keys[keyId] = {publicKey, privateKey, metadata};
83
84 return true;
85 } catch (const std::exception& e) {
86 lastError = e.what();
87 return false;
88 }
89 }

91 bool retrieveKey(const std::string& keyId,
92 std::vector<uint8_t>& publicKey,
93 std::vector<uint8_t>& privateKey,
94 KeyMetadata& metadata) {
95 try {
96 std::lock_guard<std::mutex> lock(mutex);
97
98 auto it = keys.find(keyId);
99 if (it == keys.end()) {
100 throw std::runtime_error("Key with ID " + keyId + " not found");
101 }
102
103 // Check if key is expired
104 if (std::chrono::system_clock::now() > it->second.metadata.expirationTime) {
105 throw std::runtime_error("Key with ID " + keyId + " has expired");
106 }
107
108 // Check if key is active
109 if (!it->second.metadata.isActive) {
110 throw std::runtime_error("Key with ID " + keyId + " is not active");
111 }
112
113 publicKey = it->second.publicKey;
114 privateKey = it->second.privateKey;
115 metadata = it->second.metadata;
116
117 return true;
118 } catch (const std::exception& e) {
119 lastError = e.what();
120 return false;
121 }
122 }

124 bool rotateKey(const std::string& keyId,
125 MLKEM::SecurityLevel newKemLevel,
126 MLDSA::SecurityLevel newDsaLevel,
127 const std::chrono::seconds& newExpirationTime) {
128 try {
129 std::lock_guard<std::mutex> lock(mutex);
130
131 auto it = keys.find(keyId);
132 if (it == keys.end()) {
133 throw std::runtime_error("Key with ID " + keyId + " not found");
134 }
135
136 // Generate new key pair
137 HybridCrypto crypto(newKemLevel, newDsaLevel);
138 std::vector<uint8_t> newPublicKey, newPrivateKey;
139 if (!crypto.rotateKeys(it->second.privateKey, newPublicKey, newPrivateKey)) {
140 throw std::runtime_error("Failed to rotate key: " +
141 crypto.getLastError());
142 }
143
144 // Update key and metadata
145 it->second.publicKey = newPublicKey;
146 it->second.privateKey = newPrivateKey;
147 it->second.metadata.creationTime = std::chrono::system_clock::now();
148 it->second.metadata.expirationTime = it->second.metadata.creationTime +
149 newExpirationTime;
150 it->second.metadata.algorithm = "ML-KEM-" +
151 std::to_string(static_cast<int>(newKemLevel)) +
152 "+ML-DSA-" +
153 std::to_string(static_cast<int>(newDsaLevel));
154 it->second.metadata.securityLevel = std::max(static_cast<int>(newKemLevel),
155 static_cast<int>(newDsaLevel));
156
157 return true;

159 lastError = e.what();
