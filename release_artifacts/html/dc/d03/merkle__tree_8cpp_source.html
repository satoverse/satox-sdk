40void MerkleTree::buildTree(const std::vector<std::string>& transactions) {
41 if (transactions.empty()) {
42 root_ = nullptr;
43 return;
44 }
45
46 // Create leaf nodes in parallel
47 const int numThreads = std::thread::hardware_concurrency();
48 std::vector<std::shared_ptr<Node>> nodes(transactions.size());
49 std::vector<std::future<void>> futures;
50
51 auto processChunk = [&](size_t start, size_t end) {
52 for (size_t i = start; i < end; i++) {
53 nodes[i] = std::make_shared<Node>(calculateHash(transactions[i]));
54 nodes[i]->data = transactions[i];
55 }
56 };
57
58 // Split work among threads
59 size_t chunkSize = transactions.size() / numThreads;
60 for (int i = 0; i < numThreads; i++) {
61 size_t start = i * chunkSize;
62 size_t end = (i == numThreads - 1) ? transactions.size() : (i + 1) * chunkSize;
63 futures.push_back(std::async(std::launch::async, processChunk, start, end));
64 }
65
66 // Wait for all threads to complete
67 for (auto& future : futures) {
68 future.wait();
69 }
70
71 // Build the tree
72 buildTreeRecursive(nodes);
73 root_ = nodes[0];
74}

110std::string MerkleTree::calculateHashOptimized(const std::string& data) {
111 unsigned char hash[SHA256_DIGEST_LENGTH];
112 SHA256_CTX sha256;
113 SHA256_Init(&sha256);
114
115 // Process data in chunks for better cache utilization
116 const size_t chunkSize = 64; // SHA-256 block size
117 size_t remaining = data.size();
118 const char* ptr = data.c_str();
119
120 while (remaining >= chunkSize) {
121 SHA256_Update(&sha256, ptr, chunkSize);
122 ptr += chunkSize;
123 remaining -= chunkSize;
124 }
125
126 if (remaining > 0) {
127 SHA256_Update(&sha256, ptr, remaining);
128 }
129
130 SHA256_Final(hash, &sha256);
131
132 // Optimize hex conversion with SIMD
133 std::string result;
134 result.reserve(SHA256_DIGEST_LENGTH * 2);
135
136 const char* hex = "0123456789abcdef";
137 for (int i = 0; i < SHA256_DIGEST_LENGTH; i++) {
138 result += hex[(hash[i] >> 4) & 0xF];
139 result += hex[hash[i] & 0xF];
140 }
141
142 return result;
143}

157void MerkleTree::buildTreeRecursive(std::vector<std::shared_ptr<Node>>& nodes) {
158 if (nodes.size() == 1) {
159 return;
160 }
161
162 // Process nodes in parallel
163 const int numThreads = std::thread::hardware_concurrency();
164 std::vector<std::shared_ptr<Node>> newLevel((nodes.size() + 1) / 2);
165 std::vector<std::future<void>> futures;
166
167 auto processChunk = [&](size_t start, size_t end) {
168 for (size_t i = start; i < end; i++) {
169 size_t leftIdx = i * 2;
170 size_t rightIdx = leftIdx + 1;
171
172 if (rightIdx < nodes.size()) {
173 // Create parent node from two children
174 newLevel[i] = std::make_shared<Node>(
175 combineHashesOptimized(nodes[leftIdx]->hash, nodes[rightIdx]->hash)
176 );
177 newLevel[i]->left = nodes[leftIdx];
178 newLevel[i]->right = nodes[rightIdx];
179 } else {
180 // Handle odd number of nodes by duplicating the last node
181 newLevel[i] = std::make_shared<Node>(
182 combineHashesOptimized(nodes[leftIdx]->hash, nodes[leftIdx]->hash)
183 );
184 newLevel[i]->left = nodes[leftIdx];
185 newLevel[i]->right = nodes[leftIdx];
186 }
187 }
188 };
189
190 // Split work among threads
191 size_t chunkSize = newLevel.size() / numThreads;
192 for (int i = 0; i < numThreads; i++) {
193 size_t start = i * chunkSize;
194 size_t end = (i == numThreads - 1) ? newLevel.size() : (i + 1) * chunkSize;
195 futures.push_back(std::async(std::launch::async, processChunk, start, end));
196 }
197
198 // Wait for all threads to complete
199 for (auto& future : futures) {
200 future.wait();
201 }
202
203 nodes = std::move(newLevel);
204 buildTreeRecursive(nodes);
205}

207std::vector<std::string> MerkleTree::getProofRecursive(const std::shared_ptr<Node>& node,
208 const std::string& transaction,
209 std::vector<std::string>& proof) const {
210 if (!node) {
211 return {};
212 }
213
214 if (node->data == transaction) {
215 return proof;
216 }
217
218 // Use binary search to find the transaction
219 if (node->left) {
220 proof.push_back(node->right->hash);
221 auto leftProof = getProofRecursive(node->left, transaction, proof);
222 if (!leftProof.empty()) {
223 return leftProof;
224 }
225 proof.pop_back();
226 }
227
228 if (node->right) {
229 proof.push_back(node->left->hash);
230 auto rightProof = getProofRecursive(node->right, transaction, proof);
231 if (!rightProof.empty()) {
232 return rightProof;
233 }
234 proof.pop_back();
235 }
236
237 return {};
238}
