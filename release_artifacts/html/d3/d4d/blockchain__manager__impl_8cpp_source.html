102 void startSync() {
103 std::lock_guard<std::mutex> lock(mutex_);
104
105 if (!initialized_ || !connected_) {
106 spdlog::error("BlockchainManager not initialized or not connected");
107 return;
108 }
109
110 if (is_syncing_) {
111 return;
112 }
113
114 is_syncing_ = true;
115 chain_state_.isSyncing = true;
116
117 sync_thread_ = std::thread([this]() {
118 while (is_syncing_) {
119 try {
120 // Get latest block height from network
121 uint64_t network_height = getNetworkHeight();
122
123 if (network_height > current_height_) {
124 spdlog::info("Syncing blocks from height {} to {}", current_height_ + 1, network_height);
125
126 // Request missing blocks
127 for (uint64_t height = current_height_ + 1; height <= network_height; height++) {
128 if (!is_syncing_) break;
129
130 // Request block
131 NetworkMessage msg;
132 msg.type = MessageType::GET_BLOCK_BY_HEIGHT;
133 msg.data = std::to_string(height);
134 network_->sendMessage(msg);
135
136 // Wait for response
137 std::this_thread::sleep_for(std::chrono::milliseconds(100));
138 }
139 }
140
141 // Process any pending network messages
142 processNetworkMessages();
143
144 // Sleep to prevent excessive CPU usage
145 std::this_thread::sleep_for(std::chrono::milliseconds(1000));
146 } catch (const std::exception& e) {
147 spdlog::error("Sync error: {}", e.what());
148 std::this_thread::sleep_for(std::chrono::seconds(5)); // Wait before retrying
149 }
150 }

169 bool validateBlock(const Block& block) const {
170 if (!initialized_) {
171 spdlog::error("BlockchainManager not initialized");
172 return false;
173 }
174
175 // Validate block structure
176 if (block.hash.empty() || block.previousHash.empty() || block.transactions.empty()) {
177 spdlog::error("Invalid block structure");
178 return false;
179 }
180
181 // Validate block proof
182 if (!verifyBlockProof(block)) {
183 spdlog::error("Invalid block proof");
184 return false;
185 }
186
187 // Validate transactions
188 for (const auto& tx : block.transactions) {
189 if (!validateTransaction(tx)) {
190 spdlog::error("Invalid transaction in block");
191 return false;
192 }
193 }
194
195 return true;

232 void handleNetworkMessage(const NetworkMessage& message) {
233 switch (message.type) {
234 case MessageType::BLOCK: {
235 Block block = deserializeBlock(message.data);
236 if (validateBlock(block)) {
237 std::lock_guard<std::mutex> lock(mutex_);
238 if (block.height == current_height_ + 1) {
239 // Add block to chain
240 saveBlock(block);

242
243 // Notify listeners
244 for (const auto& callback : sync_callbacks_) {
245 callback(block);
246 }
247 }
248 }
249 break;
250 }

252 spdlog::error("Network error: {}", message.data);
