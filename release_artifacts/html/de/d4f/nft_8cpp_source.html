50NFT NFTManager::mintNFT(
51 const std::string& contractAddress,
52 const std::string& creator,
53 const NFTMetadata& metadata,
54 const std::string& uri) {
55
56 if (!initialized_) {
57 throw std::runtime_error("NFTManager not initialized");
58 }
59
60 if (contractAddress.empty() || creator.empty() || metadata.name.empty()) {
61 throw std::runtime_error("Contract address, creator, and metadata name are required");
62 }
63
64 NFT nft;
65 nft.contractAddress = contractAddress;
66 nft.creator = creator;
67 nft.owner = creator; // Initially owned by creator
68 nft.metadata = metadata;
69 nft.uri = uri;
70 nft.createdAt = std::chrono::system_clock::now();
71 nft.status = "active";
72
73 // Generate token ID and NFT ID
74 nft.tokenId = generateTokenId(contractAddress);
75 nft.id = generateNFTId(nft);
76
77 // Initialize balance for creator
78 balances_[contractAddress][creator] = "1";
79
80 // Store NFT
81 nfts_[nft.id] = nft;
82
83 return nft;

255std::string NFTManager::generateNFTId(const NFT& nft) {
256 std::stringstream ss;
257 ss << nft.contractAddress << nft.tokenId << nft.creator
258 << nft.createdAt.time_since_epoch().count();
259
260 // Generate SHA-256 hash
261 unsigned char hash[SHA256_DIGEST_LENGTH];
262 SHA256_CTX sha256;
263 SHA256_Init(&sha256);
264 SHA256_Update(&sha256, ss.str().c_str(), ss.str().size());
265 SHA256_Final(hash, &sha256);
266
267 // Convert to hex string
268 std::stringstream hexStream;
269 for (int i = 0; i < SHA256_DIGEST_LENGTH; i++) {
270 hexStream << std::hex << std::setw(2) << std::setfill('0') << static_cast<int>(hash[i]);
271 }
272
273 return hexStream.str();
274}
