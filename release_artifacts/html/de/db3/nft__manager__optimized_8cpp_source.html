43class ThreadPool {
44public:
45 ThreadPool(size_t threads) : stop(false) {
46 for(size_t i = 0; i < threads; ++i)
47 workers.emplace_back([this] {
48 while(true) {
49 std::function<void()> task;
50 {
51 std::unique_lock<std::mutex> lock(this->queue_mutex);
52 this->condition.wait(lock, [this] {
53 return this->stop || !this->tasks.empty();
54 });
55 if(this->stop && this->tasks.empty())
56 return;
57 task = std::move(this->tasks.front());
58 this->tasks.pop();
59 }
60 task();
61 }
62 });
63 }
64
65 template<class F>
66 std::future<void> enqueue(F&& f) {
67 std::packaged_task<void()> task(std::forward<F>(f));
68 std::future<void> res = task.get_future();
69 {
70 std::unique_lock<std::mutex> lock(queue_mutex);
71 if(stop)
72 throw std::runtime_error("enqueue on stopped ThreadPool");
73 tasks.emplace(std::move(task));
74 }
75 condition.notify_one();
76 return res;
77 }
78
79 ~ThreadPool() {
80 {
81 std::unique_lock<std::mutex> lock(queue_mutex);
82 stop = true;
83 }
84 condition.notify_all();
85 for(std::thread &worker: workers)
86 worker.join();
87 }
88
89private:
90 std::vector<std::thread> workers;
91 std::queue<std::function<void()>> tasks;
92 std::mutex queue_mutex;
93 std::condition_variable condition;
94 bool stop;
95};

197 std::lock_guard<std::mutex> lock(mutex_);
198 if (!initialized_) {
199 throw std::runtime_error("NFTManager not initialized");
200 }

206 // Store metadata in IPFS

208 std::string metadataCid = storeInIPFS(metadataJson);

210 // Create NFT on blockchain
211 NFT nftCopy = nft;

213 nfts_[nft.id] = nftCopy;

216 ownerNFTs_[nft.owner].push_back(nft.id);
217 if (!nft.collection.empty()) {

240 NFT getNFT(const std::string& nftId) {
241 {
242 std::shared_lock<std::shared_mutex> lock(mutex_);
243 auto cacheIt = cache_.find(nftId);
244 if (cacheIt != cache_.end()) {
245 cacheIt->second.lastAccessed = std::chrono::system_clock::now();
246 return cacheIt->second.nft;
247 }
248 }
249
250 std::unique_lock<std::shared_mutex> lock(mutex_);
251 auto it = nfts_.find(nftId);
252 if (it == nfts_.end()) {
253 throw std::runtime_error("NFT not found");
254 }
255
256 // Update cache
257 CacheEntry entry{it->second, std::chrono::system_clock::now()};
258 cache_[nftId] = entry;
259
260 // Implement cache size limit
261 if (cache_.size() > max_cache_size_) {
262 cleanupCache();
263 }

283 bool transferNFT(const std::string& nftId, const std::string& from, const std::string& to) {
284 std::unique_lock<std::shared_mutex> lock(mutex_);
285 auto it = nfts_.find(nftId);
286 if (it == nfts_.end()) {
287 return false;
288 }
289
290 if (it->second.owner != from) {
291 return false;
292 }
293
294 // Update NFT
295 it->second.owner = to;
296
297 // Update indices
298 auto& fromNFTs = ownerNFTs_[from];
299 fromNFTs.erase(std::remove(fromNFTs.begin(), fromNFTs.end(), nftId), fromNFTs.end());
300 ownerNFTs_[to].push_back(nftId);
301
302 // Update cache
303 auto cacheIt = cache_.find(nftId);
304 if (cacheIt != cache_.end()) {
305 cacheIt->second.nft = it->second;
306 cacheIt->second.lastAccessed = std::chrono::system_clock::now();
307 }
308
309 // Notify listeners asynchronously
310 notifyListenersAsync(it->second, "transferred");
311
312 return true;
