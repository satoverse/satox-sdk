73std::shared_ptr<Transaction> Transaction::fromJson(const nlohmann::json& json) {
74 auto tx = std::make_shared<Transaction>();
75
76 tx->setHash(json["hash"].get<std::string>());
77 tx->setFrom(json["from"].get<std::string>());
78 tx->setTo(json["to"].get<std::string>());
79 tx->setValue(json["value"].get<double>());
80 tx->setTimestamp(std::chrono::system_clock::from_time_t(json["timestamp"].get<time_t>()));
81 tx->setStatus(json["status"].get<std::string>());
82 tx->setMetadata(json["metadata"]);
83
84 // Convert hex string to binary data
85 std::string hex_data = json["data"].get<std::string>();
86 std::vector<uint8_t> data;
87 for (size_t i = 0; i < hex_data.length(); i += 2) {
88 std::string byteString = hex_data.substr(i, 2);
89 uint8_t byte = static_cast<uint8_t>(std::stoi(byteString, nullptr, 16));
90 data.push_back(byte);
91 }
92 tx->setData(data);
93
94 return tx;
95}
