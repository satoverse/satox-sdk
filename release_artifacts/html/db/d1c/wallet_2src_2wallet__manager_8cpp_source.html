743 // Finalize encryption

745 EVP_CIPHER_CTX_free(ctx);
746 setError("Failed to finalize encryption");
747 return false;
748 }
749 ciphertext_len += len;
750
751 // Get authentication tag
752 std::vector<uint8_t> tag(16);
753 if (EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_GCM_GET_TAG, tag.size(), tag.data()) != 1) {
754 EVP_CIPHER_CTX_free(ctx);
755 setError("Failed to get authentication tag");
756 return false;
757 }
758
759 // Cleanup
760 EVP_CIPHER_CTX_free(ctx);
761
762 // Combine salt, IV, ciphertext, and tag
763 encrypted.clear();
764 encrypted.insert(encrypted.end(), salt.begin(), salt.end());
765 encrypted.insert(encrypted.end(), iv.begin(), iv.end());
766 encrypted.insert(encrypted.end(), ciphertext.begin(), ciphertext.begin() + ciphertext_len);
767 encrypted.insert(encrypted.end(), tag.begin(), tag.end());
768
769 return true;
770 } catch (const std::exception& e) {
771 setError("Private key encryption failed: " + std::string(e.what()));
772 return false;
773 }

776bool WalletManager::decryptPrivateKey(const std::vector<uint8_t>& encrypted, const std::string& password, std::vector<uint8_t>& privateKey) {
777 try {
778 if (encrypted.size() < 44) { // 16 (salt) + 12 (IV) + 16 (tag) minimum
779 setError("Invalid encrypted data size");
780 return false;
781 }
782
783 // Extract components
784 std::vector<uint8_t> salt(encrypted.begin(), encrypted.begin() + 16);
785 std::vector<uint8_t> iv(encrypted.begin() + 16, encrypted.begin() + 28);
786 std::vector<uint8_t> tag(encrypted.end() - 16, encrypted.end());
787 std::vector<uint8_t> ciphertext(encrypted.begin() + 28, encrypted.end() - 16);
788
789 // Derive key from password
790 std::vector<uint8_t> key(32); // AES-256 key
791 if (PKCS5_PBKDF2_HMAC(
792 password.c_str(), password.length(),
793 salt.data(), salt.size(),
794 100000, // Same iteration count as encryption
795 EVP_sha256(),
796 key.size(), key.data()) != 1) {
797 setError("Failed to derive decryption key");
798 return false;
799 }
800
801 // Initialize decryption context
802 EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
803 if (!ctx) {
804 setError("Failed to create decryption context");
805 return false;
806 }
807
808 // Initialize decryption
809 if (EVP_DecryptInit_ex(ctx, EVP_aes_256_gcm(), nullptr, key.data(), iv.data()) != 1) {
810 EVP_CIPHER_CTX_free(ctx);
811 setError("Failed to initialize decryption");
812 return false;
813 }
814
815 // Add associated data (salt)
816 int len;
817 if (EVP_DecryptUpdate(ctx, nullptr, &len, salt.data(), salt.size()) != 1) {
818 EVP_CIPHER_CTX_free(ctx);
819 setError("Failed to add associated data");
820 return false;
821 }
822
823 // Decrypt private key
824 privateKey.resize(ciphertext.size());
825 if (EVP_DecryptUpdate(ctx, privateKey.data(), &len, ciphertext.data(), ciphertext.size()) != 1) {
826 EVP_CIPHER_CTX_free(ctx);
827 setError("Failed to decrypt private key");
828 return false;
829 }
830 int plaintext_len = len;
831
832 // Set expected tag
833 if (EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_GCM_SET_TAG, tag.size(), tag.data()) != 1) {
834 EVP_CIPHER_CTX_free(ctx);
835 setError("Failed to set authentication tag");
836 return false;
837 }
838
839 // Finalize decryption
840 if (EVP_DecryptFinal_ex(ctx, privateKey.data() + len, &len) != 1) {
841 EVP_CIPHER_CTX_free(ctx);
842 setError("Failed to finalize decryption");
843 return false;
844 }
845 plaintext_len += len;
846
847 // Cleanup
848 EVP_CIPHER_CTX_free(ctx);
849
850 // Resize to actual length
851 privateKey.resize(plaintext_len);
852 return true;
853 } catch (const std::exception& e) {
854 setError("Private key decryption failed: " + std::string(e.what()));
855 return false;
856 }
