124void SecurityMonitor::recordOperation(const std::string& operation,
125 const std::chrono::milliseconds& duration,
126 bool success) {
127 auto& metrics = operationMetrics[operation];
128 metrics.totalOperations++;
129 if (success) {
130 metrics.successfulOperations++;
131 } else {
132 metrics.failedOperations++;
133 }
134 metrics.totalDuration += duration.count();
135
136 {
137 std::lock_guard<std::mutex> lock(metrics.mutex);
138 metrics.recentDurations.push_back(duration);
139 if (metrics.recentDurations.size() > 1000) {
140 metrics.recentDurations.erase(metrics.recentDurations.begin());
141 }
142 }
143
144 // Update Prometheus metrics
145 if (operationCounters.count(operation)) {
146 operationCounters[operation]->Increment();
147 }
148 if (operationDurations.count(operation)) {
149 operationDurations[operation]->Observe(duration.count() / 1000.0);
150 }
151}

210double SecurityMonitor::getAverageOperationTime(const std::string& operation) const {
211 auto it = operationMetrics.find(operation);
212 if (it == operationMetrics.end()) {
213 return 0.0;
214 }
215
216 const auto& metrics = it->second;
217 if (metrics.totalOperations == 0) {
218 return 0.0;
219 }
220
221 return static_cast<double>(metrics.totalDuration) / metrics.totalOperations;
222}

310void SecurityMonitor::updatePrometheusMetrics() {
311 // Update operation metrics
312 for (const auto& [operation, metrics] : operationMetrics) {
313 if (operationCounters.count(operation)) {
314 operationCounters[operation]->Increment(metrics.totalOperations);
315 }
316 if (operationDurations.count(operation)) {
317 operationDurations[operation]->Observe(
318 static_cast<double>(metrics.totalDuration) / 1000.0
319 );
320 }
321 }
322
323 // Update resource metrics
324 memoryGauge->Set(resourceMetrics.currentMemoryUsage);
325 cpuGauge->Set(resourceMetrics.currentCpuUsage);
326
327 // Update cache metrics
328 cacheHitCounter->Increment(cacheMetrics.hits);
329 cacheMissCounter->Increment(cacheMetrics.misses);
330
331 // Update error metrics
332 {
333 std::lock_guard<std::mutex> lock(errorMetrics.mutex);
334 for (const auto& [error, count] : errorMetrics.errorCounts) {
335 if (errorCounters.count(error)) {
336 errorCounters[error]->Increment(count);
337 }
338 }
339 }
