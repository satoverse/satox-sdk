72 return false;
73 }
74
75 try {
76 nlohmann::json nftJson = {
77 {"id", nft.id},
78 {"contractAddress", nft.contractAddress},

83 {"isTransferable", nft.isTransferable},

88

168
169 // Check cache first

194 nft.royaltyRecipient = nftJson["royaltyRecipient"];
195 nft.royaltyBasisPoints = nftJson["royaltyBasisPoints"];
196
197 if (cacheEnabled_) {
198 nftCache_[nftId] = nft;
199 if (nftCache_.size() > maxCacheSize_) {
200 nftCache_.erase(nftCache_.begin());

386std::vector<NFTManager::NFT> NFTStorage::searchNFTs(const std::string& query) {
387 std::lock_guard<std::mutex> lock(mutex_);
388 std::vector<NFTManager::NFT> results;
389
390 // Search in owner index
391 for (const auto& [owner, nftIds] : ownerIndex_) {
392 if (owner.find(query) != std::string::npos) {
393 for (const auto& nftId : nftIds) {
394 if (auto nft = getNFT(nftId)) {
395 results.push_back(*nft);
396 }
397 }
398 }
399 }
400
401 // Search in contract index
402 for (const auto& [contract, nftIds] : contractIndex_) {
403 if (contract.find(query) != std::string::npos) {
404 for (const auto& nftId : nftIds) {
405 if (auto nft = getNFT(nftId)) {
406 results.push_back(*nft);
407 }
408 }
409 }
410 }
411
412 // Remove duplicates
413 std::sort(results.begin(), results.end(), [](const NFTManager::NFT& a, const NFTManager::NFT& b) {
414 return a.id < b.id;
415 });
416 results.erase(std::unique(results.begin(), results.end(), [](const NFTManager::NFT& a, const NFTManager::NFT& b) {
417 return a.id == b.id;
418 }), results.end());
419
420 return results;
421}

423std::vector<NFTManager::NFT> NFTStorage::searchNFTsByMetadata(const nlohmann::json& metadataQuery) {
424 std::lock_guard<std::mutex> lock(mutex_);
425 std::vector<NFTManager::NFT> results;
426
427 // Search through all NFTs
428 for (const auto& [nftId, _] : nftCache_) {
429 if (auto metadata = getNFTMetadata(nftId)) {
430 bool matches = true;
431 for (const auto& [key, value] : metadataQuery.items()) {
432 if (!metadata->attributes.empty() &&
433 std::find(metadata->attributes.begin(), metadata->attributes.end(), value.get<std::string>()) == metadata->attributes.end()) {
434 matches = false;
435 break;
436 }
437 }
438 if (matches) {
439 if (auto nft = getNFT(nftId)) {
440 results.push_back(*nft);
441 }
442 }
443 }
444 }
445
446 return results;
447}

651bool NFTStorage::removeFromIndex(const std::string& nftId) {
652 try {
653 // Remove from owner index
654 for (auto& [owner, nftIds] : ownerIndex_) {
655 nftIds.erase(std::remove(nftIds.begin(), nftIds.end(), nftId), nftIds.end());

664 if (nftIds.empty()) {

668
669 // Remove from creator index

671 nftIds.erase(std::remove(nftIds.begin(), nftIds.end(), nftId), nftIds.end());
672 if (nftIds.empty()) {
