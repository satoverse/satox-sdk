48bool SatoxManager::initialize(const nlohmann::json& config) {
49 std::lock_guard<std::recursive_mutex> lock(mutex_);
50
51 if (initialized_) {
52 lastError_ = "SatoxManager already initialized";
53 return false;
54 }
55
56 config_ = config;
57
58 // Initialize core components
59 if (!initializeCoreComponent()) {
60 return false;

73 }
74
75 if (!initializeIPFSComponent()) {
76 return false;
77 }
78
79 if (!initializeNetworkComponent()) {
80 return false;
81 }
82
83 if (!initializeConfigComponent()) {
84 return false;
85 }
86
87 initialized_ = true;
88 stats_.start_time = std::chrono::system_clock::now();
89 stats_.last_activity = std::chrono::system_clock::now();
90
91 return true;
