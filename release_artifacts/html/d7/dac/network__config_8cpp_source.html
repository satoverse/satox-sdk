53bool NetworkConfigManager::loadConfig(const std::string& configPath) {
54 try {
55 if (!std::filesystem::exists(configPath)) {
56 // Create default config file if it doesn't exist
57 return saveConfig(configPath);
58 }
59
60 std::ifstream file(configPath);
61 if (!file.is_open()) {
62 return false;
63 }
64
65 nlohmann::json j;
66 file >> j;
67
68 currentConfig_.type = stringToNetworkType(j["network_type"]);
69 currentConfig_.rpcHost = j["rpc_host"];
70 currentConfig_.rpcPort = j["rpc_port"];
71 currentConfig_.rpcUser = j["rpc_user"];
72 currentConfig_.rpcPassword = j["rpc_password"];
73 currentConfig_.useSSL = j["use_ssl"];
74 currentConfig_.timeoutSeconds = j["timeout_seconds"];
75 currentConfig_.retryAttempts = j["retry_attempts"];
76
77 isInitialized_ = true;
78 return true;
79 } catch (const std::exception& e) {
80 return false;
81 }

84bool NetworkConfigManager::saveConfig(const std::string& configPath) {
85 try {
86 // Create directory if it doesn't exist
87 std::filesystem::create_directories(
88 std::filesystem::path(configPath).parent_path());
89
90 std::ofstream file(configPath);
91 if (!file.is_open()) {
92 return false;
93 }
94
95 nlohmann::json j;
96 j["network_type"] = networkTypeToString(currentConfig_.type);
97 j["rpc_host"] = currentConfig_.rpcHost;
98 j["rpc_port"] = currentConfig_.rpcPort;
99 j["rpc_user"] = currentConfig_.rpcUser;
100 j["rpc_password"] = currentConfig_.rpcPassword;
101 j["use_ssl"] = currentConfig_.useSSL;
102 j["timeout_seconds"] = currentConfig_.timeoutSeconds;
103 j["retry_attempts"] = currentConfig_.retryAttempts;
104
105 file << j.dump(4);
106 return true;

108 return false;
