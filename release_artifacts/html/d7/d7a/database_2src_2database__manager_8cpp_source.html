110void DatabaseManager::shutdown() {
111 std::cerr << "[DEBUG] DatabaseManager::shutdown() called" << std::endl;
112 std::lock_guard<std::recursive_mutex> lock(mutex_);
113
114 if (!initialized_.load()) {
115 return;
116 }
117
118 // Mark as not initialized first to prevent new operations
119 initialized_.store(false);
120
121 try {
122 // Disconnect first if connected
123 if (isConnected_) {
124 try {
125 disconnect();
126 } catch (const std::exception& e) {
127 // Ignore disconnect errors during shutdown
128 }
129 }
130
131 // Clear callbacks before clearing other data
132 try {
133 databaseCallback_ = nullptr;
134 connectionCallback_ = nullptr;
135 healthCallback_ = nullptr;
136 } catch (const std::exception& e) {
137 // Ignore callback clearing errors during shutdown
138 }
139
140 // Clear all data structures safely
141 try {
142 currentDatabase_.clear();
143 tableSchemas_.clear();
144 tableIndexes_.clear();
145 databases_.clear();
146 transactionSnapshot_.clear();
147 } catch (const std::exception& e) {
148 // Ignore data structure clearing errors during shutdown
149 }
150
151 // Update stats without logging to avoid logger issues
152 try {
153 stats_.totalOperations++;
154 stats_.successfulOperations++;
155 stats_.lastOperation = std::chrono::system_clock::now();
156 } catch (const std::exception& e) {
157 // Ignore stats update errors during shutdown
158 }
159
160 // Safely drop all loggers first
161 try {
162 if (logger_) {
163 logger_.reset();
164 }
165 spdlog::drop_all();
166 } catch (const std::exception& e) {
167 // Ignore logger cleanup errors
168 }
169
170 } catch (const std::exception& e) {
171 std::cerr << "Error during DatabaseManager shutdown: " << e.what() << std::endl;
172 // Ensure logger is reset even if there's an error
173 try {
174 logger_.reset();
175 spdlog::drop_all();
176 } catch (...) {
177 // Ignore all errors during final cleanup
178 }
179 }
180}

621 std::lock_guard<std::recursive_mutex> lock(mutex_);
622
623 // if (logger_) logger_->debug("DatabaseManager::update() - ENTRY: table={}, id={}", tableName, id);
624

626 // if (logger_) logger_->debug("DatabaseManager::update() - EXIT (not initialized or no current database)");
