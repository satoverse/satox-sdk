127 try {
128 if (!config["security"]["pqc"]["enabled"].get<bool>()) {
129 return fallbackEncrypt(data, encryptedData);
130 }
131
132 // Check operation cache
133 std::string cacheKey = generateCacheKey("encrypt", data);
134 if (auto cached = operationCache->get(cacheKey)) {
135 if (!isCacheEntryExpired(*cached)) {
136 encryptedData = cached->result;
137 return true;
138 }
139 }
140
141 // Get current key pair with caching
142 std::string keyId = "default_key";
143 KeyCacheEntry keyEntry;
144 {
145 std::shared_lock<std::shared_mutex> lock(cacheMutex);
146 if (auto cached = keyCache->get(keyId)) {
147 if (!isCacheEntryExpired(*cached)) {
148 keyEntry = *cached;
149 }
150 }
151 }
152
153 if (keyEntry.keys.empty()) {
154 std::vector<uint8_t> publicKey, privateKey;
155 pqc::KeyMetadata metadata;

157 if (!keyManager->retrieveKey(keyId, publicKey, privateKey, metadata)) {
158 return fallbackEncrypt(data, encryptedData);
159 }

161 keyEntry.keys = {publicKey, privateKey};
162 keyEntry.metadata = metadata;
163 keyEntry.timestamp = std::chrono::system_clock::now();
164
165 std::unique_lock<std::shared_mutex> lock(cacheMutex);
166 keyCache->put(keyId, keyEntry);

330 bool parallelEncrypt(const std::vector<uint8_t>& data,
331 const std::vector<uint8_t>& publicKey,
332 std::vector<uint8_t>& encryptedData) {
333 const size_t chunkSize = 1024 * 1024; // 1MB chunks
334 const size_t numChunks = (data.size() + chunkSize - 1) / chunkSize;
335 std::vector<std::vector<uint8_t>> encryptedChunks(numChunks);
336 std::vector<std::future<bool>> futures;
337
338 for (size_t i = 0; i < numChunks; ++i) {
339 size_t start = i * chunkSize;
340 size_t end = std::min(start + chunkSize, data.size());
341 std::vector<uint8_t> chunk(data.begin() + start, data.begin() + end);
342
343 futures.push_back(threadPool->submit([this, chunk, &publicKey, &encryptedChunks, i]() {
344 return hybridCrypto->hybridEncrypt(publicKey, chunk, encryptedChunks[i]);
345 }));
346 }
347
348 bool success = true;
349 for (auto& future : futures) {
350 success &= future.get();
351 }
352
353 if (success) {
354 // Combine encrypted chunks
355 size_t totalSize = 0;
356 for (const auto& chunk : encryptedChunks) {
357 totalSize += chunk.size();
358 }
359 encryptedData.reserve(totalSize);
360 for (const auto& chunk : encryptedChunks) {
361 encryptedData.insert(encryptedData.end(), chunk.begin(), chunk.end());
362 }
363 }
364
365 return success;
366 }
