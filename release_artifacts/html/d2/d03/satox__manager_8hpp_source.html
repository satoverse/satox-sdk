232// Template implementations
233template<typename T>
234bool SatoxManager::registerComponent(const std::string& name, std::shared_ptr<T> component) {
235 if (!component) {
236 setError("Invalid component pointer");
237 return false;
238 }
239
240 std::lock_guard<std::recursive_mutex> lock(mutex_);
241 if (components_.find(name) != components_.end()) {
242 setError("Component already registered: " + name);
243 return false;
244 }
245
246 components_[name] = std::static_pointer_cast<void>(component);
247 return true;
248}
249
250template<typename T>
251std::shared_ptr<T> SatoxManager::getComponent(const std::string& name) {
252 std::lock_guard<std::recursive_mutex> lock(mutex_);
253 auto it = components_.find(name);
254 if (it == components_.end()) {
255 setError("Component not found: " + name);
256 return nullptr;
257 }
258
259 return std::static_pointer_cast<T>(it->second);
260}
